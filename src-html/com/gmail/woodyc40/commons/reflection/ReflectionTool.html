017package com.gmail.woodyc40.commons.reflection;
018
019import com.gmail.woodyc40.commons.misc.Checks;
020
021import java.lang.reflect.*;
022
023/**
024 * Utility methods to fetch fields and checked Reflection using {@link com.gmail.woodyc40.commons.reflection} package
025 *
026 * @author AgentTroll
027 * @version 1.0
028 * @see com.gmail.woodyc40.commons.reflection.ConstructorManager
029 * @see com.gmail.woodyc40.commons.reflection.FieldManager
030 * @see com.gmail.woodyc40.commons.reflection.MethodManager
031 * @since 1.0
032 */
033public final class ReflectionTool {
034    private ReflectionTool() {}
035
036    /**
037     * Makes a {@code class} search for a field by name
038     *
039     * @param name   name of the field declared by the {@code class}
040     * @param holder the {@code class} containing the field
041     * @return the {@link java.lang.reflect.Field} object held by name
042     */
043    public static Field forField(String name, Class<?> holder) {
044        Checks.notNull(name, "Cannot find null field");
045        Checks.notNull(holder, "Cannot find " + name + " in class null");
046
047        try {
048            return holder.getDeclaredField(name);
049        } catch (NoSuchFieldException e) {
050            e.printStackTrace();
051        }
052
053        return null;
054    }
055
056    /**
057     * Makes a {@code class} and all of its superclass BUT {@link java.lang.Object}.class search for a field by name
058     *
059     * @param name   name of the field declared by the {@code class}
060     * @param holder the {@code class} containing the field
061     * @return the {@link java.lang.reflect.Field} object held by name
062     */
063    public static Field forFieldDeep(String name, Class<?> holder) {
064        Checks.notNull(name, "Cannot find null field");
065        Checks.notNull(holder, "Cannot find " + name + " in class null");
066
067        for (Class<?> current = holder;
068             !current.getSuperclass().equals(Object.class);
069             current = current.getSuperclass()) {
070            try {
071                return current.getDeclaredField(name);
072            } catch (NoSuchFieldException e) {
073                e.printStackTrace();
074            }
075        }
076
077        return null;
078    }
079
080    /**
081     * Makes a {@code class} search for a method by name
082     *
083     * @param name   name of the method declared by the {@code class}
084     * @param holder the {@code class} containing the method
085     * @param params the parameters of the method
086     * @return the {@link java.lang.reflect.Method} object held by name
087     */
088    public static Method forMethod(String name, Class<?> holder, Class<?>... params) {
089        Checks.notNull(name, "Cannot find null field");
090        Checks.notNull(holder, "Cannot find " + name + " in class null");
091        Checks.notNull(params, "Params cannot be null, try an empty array instead");
092
093        try {
094            return holder.getDeclaredMethod(name, params);
095        } catch (NoSuchMethodException e) {
096            e.printStackTrace();
097        }
098
099        return null;
100    }
101
102    /**
103     * Makes a {@code class} search AND all superclasses until the loop reaches {@link java.lang.Object} for a method by
104     * name
105     *
106     * @param name   name of the method declared by the {@code class}
107     * @param holder the {@code class} containing the method
108     * @param params the parameters of the method
109     * @return the {@link java.lang.reflect.Method} object held by name
110     */
111    public static Method forMethodDeep(String name, Class<?> holder, Class<?>... params) {
112        Checks.notNull(name, "Cannot find null field");
113        Checks.notNull(holder, "Cannot find " + name + " in class null");
114        Checks.notNull(params, "Params cannot be null, try an empty array instead");
115
116        for (Class<?> current = holder;
117             !current.getSuperclass().equals(Object.class);
118             current = current.getSuperclass()) {
119            try {
120                return current.getDeclaredMethod(name, params);
121            } catch (NoSuchMethodException e) {
122                e.printStackTrace();
123            }
124        }
125
126        return null;
127    }
128
129    /**
130     * Makes a {@code class} search for a constructor by name
131     *
132     * @param holder the {@code class} containing the constructor
133     * @return the {@link java.lang.reflect.Constructor} object held by holder
134     */
135    public static <T> Constructor<T> forConstruct(Class<T> holder, Class<?>... params) {
136        Checks.notNull(holder, "Cannot find constructor in class null");
137        Checks.notNull(params, "Params cannot be null, try an empty array instead");
138
139        try {
140            return holder.getDeclaredConstructor(params);
141        } catch (NoSuchMethodException e) {
142            e.printStackTrace();
143        }
144
145        return null;
146    }
147
148    /**
149     * Makes a {@code class} search AND all superclasses excluding {@link java.lang.Object} for a constructor by name
150     *
151     * @param holder the {@code class} containing the constructor
152     * @return the {@link java.lang.reflect.Constructor} object held by holder
153     */
154    public static Constructor<?> forConstructDeep(Class<?> holder, Class<?>... params) {
155        Checks.notNull(holder, "Cannot find constructor in class null");
156        Checks.notNull(params, "Params cannot be null, try an empty array instead");
157
158        for (Class<?> current = holder;
159             !current.getSuperclass().equals(Object.class);
160             current = current.getSuperclass()) {
161            try {
162                return current.getDeclaredConstructor(params);
163            } catch (NoSuchMethodException e) {
164                e.printStackTrace();
165            }
166        }
167
168        return null;
169    }
170}
