017package com.gmail.woodyc40.commons.reflection.impl;
018
019import com.gmail.woodyc40.commons.reflection.MethodManager;
020import sun.reflect.MethodAccessor;
021
022import java.lang.reflect.InvocationTargetException;
023import java.lang.reflect.Method;
024
025/**
026 * Actual implementation of {@link com.gmail.woodyc40.commons.reflection.MethodManager} used for fast reflection
027 *
028 * @param <D> the {@code class} type declaring the method
029 * @param <T> the type the method returns
030 * @author AgentTroll
031 * @version 1.0
032 * @since 1.0
033 */
034class MethodImpl<D, T> implements MethodManager<D, T> {
035    private final Method         method;
036    private final MethodAccessor accessor;
037
038    /**
039     * Wraps the Method for management by this implementation
040     *
041     * @param method the Method to wrap
042     */
043    public MethodImpl(Method method) {
044        this.method = method;
045        this.accessor = ReflectAccess.getREFLECTION_FACTORY().newMethodAccessor(this.method);
046    }
047
048    @Override public T invoke(D inst, Object... args) {
049        try {
050            return (T) this.accessor.invoke(inst, args);
051        } catch (IllegalArgumentException | InvocationTargetException x) {
052            x.printStackTrace();
053        }
054
055        return null;
056    }
057
058    @Override public Method raw() {
059        return this.method;
060    }
061}
