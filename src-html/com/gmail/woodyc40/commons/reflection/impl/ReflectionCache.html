017package com.gmail.woodyc40.commons.reflection.impl;
018
019import com.gmail.woodyc40.commons.collect.*;
020import com.gmail.woodyc40.commons.reflection.*;
021
022import java.lang.reflect.*;
023
024/**
025 * Cached values for reflective access
026 *
027 * @author AgentTroll
028 * @version 1.0
029 * @since 1.0
030 */
031public final class ReflectionCache {
032    /** The cache for unsafe methods */
033    private static final ExpandableMapping<Method, MethodManager<?, ?>>           METHOD    =
034            new ExpandableMapping<>();
035    /** The cache for unsafe fields */
036    private static final ExpandableMapping<Field, FieldManager<?, ?>>             FIELD     =
037            new ExpandableMapping<>();
038    /** The cache for unsafe constructors */
039    private static final ExpandableMapping<Constructor<?>, ConstructorManager<?>> CONSTRUCT =
040            new ExpandableMapping<>();
041    /** The cache for {@code class}es */
042    private static final Cache<String, Class<?>>                                  CLASS     = new HashingCache<>();
043
044    private ReflectionCache() {} // Suppress instantiation
045
046    /**
047     * Caches a lookup for a Class
048     *
049     * @param name the FQN (fully qualified name) of the {@code class} to be looked up
050     * @return the value of the cached class, or the inserted value for a class lookup
051     */
052    public static Class<?> getClass(String name) {
053        Class<?> c = ReflectionCache.CLASS.lookup(name);
054        if (c == null) {
055            try {
056                c = Class.forName(name);
057                ReflectionCache.CLASS.insert(name, c);
058            } catch (ClassNotFoundException e) {
059                e.printStackTrace();
060            }
061        }
062
063        return c;
064    }
065
066    /**
067     * Wraps a Method with the method manager with caching properties
068     *
069     * @param method the method to wrap
070     * @param <D>    the declaring class type
071     * @param <T>    the return type
072     * @return the wrapped method
073     */
074    public static <D, T> MethodManager<D, T> method(Method method) {
075        MethodManager<D, T> cached = (MethodManager<D, T>) ReflectionCache.METHOD.get(method, "unsafe");
076        if (cached == null) {
077            cached = new MethodImpl<>(method);
078            ReflectionCache.METHOD.put(method, cached, "unsafe");
079        }
080
081        return cached;
082    }
083
084    /**
085     * Wraps a field with the field manager with caching properties
086     *
087     * @param field the field to wrap
088     * @param <D>   the declaring class type
089     * @param <T>   the field type
090     * @return the wrapped field
091     */
092    public static <D, T> FieldManager<D, T> field(Field field) {
093        FieldManager<D, T> cached = (FieldManager<D, T>) ReflectionCache.FIELD.get(field, "unsafe");
094        if (cached == null) {
095            cached = new FieldImpl<>(field);
096            ReflectionCache.FIELD.put(field, cached, "unsafe");
097        }
098
099        return cached;
100    }
101
102    /**
103     * Wraps a constructor using constructor manager with caching properties
104     *
105     * @param constructor the constructor to wrap
106     * @param <T>         the declaring class type
107     * @return the wrapped constructor
108     */
109    public static <T> ConstructorManager<T> constructor(Constructor<T> constructor) {
110        ConstructorManager<T> cached = (ConstructorManager<T>) ReflectionCache.CONSTRUCT.get(constructor, "unsafe");
111        if (cached == null) {
112            cached = new ConstructorImpl<>(constructor);
113            ReflectionCache.CONSTRUCT.put(constructor, cached, "unsafe");
114        }
115
116        return cached;
117    }
118
119    /**
120     * Wraps a Method with the method manager with caching properties
121     * <p>
122     * <p>Uses the slower {@link java.lang.reflect} classes</p>
123     *
124     * @param method the method to wrap
125     * @param <D>    the declaring class type
126     * @param <T>    the return type
127     * @return the wrapped method
128     */
129    public static <D, T> MethodManager<D, T> methodSafe(Method method) {
130        MethodManager<D, T> cached = (MethodManager<D, T>) ReflectionCache.METHOD.get(method, "safe");
131        if (cached == null) {
132            cached = new SafeMethod<>(method);
133            ReflectionCache.METHOD.put(method, cached, "safe");
134        }
135
136        return cached;
137    }
138
139    /**
140     * Wraps a field with the field manager with caching properties
141     * <p>
142     * <p>Uses the slower {@link java.lang.reflect} classes</p>
143     *
144     * @param field the field to wrap
145     * @param <D>   the declaring class type
146     * @param <T>   the field type
147     * @return the wrapped field
148     */
149    public static <D, T> FieldManager<D, T> fieldSafe(Field field) {
150        FieldManager<D, T> cached = (FieldManager<D, T>) ReflectionCache.FIELD.get(field, "safe");
151        if (cached == null) {
152            cached = new SafeField<>(field);
153            ReflectionCache.FIELD.put(field, cached, "safe");
154        }
155
156        return cached;
157    }
158
159    /**
160     * Wraps a constructor using constructor manager with caching properties
161     * <p>
162     * <p>Uses the slower {@link java.lang.reflect} classes</p>
163     *
164     * @param constructor the constructor to wrap
165     * @param <T>         the declaring class type
166     * @return the wrapped constructor
167     */
168    public static <T> ConstructorManager<T> constructorSafe(Constructor<T> constructor) {
169        ConstructorManager<T> cached = (ConstructorManager<T>) ReflectionCache.CONSTRUCT.get(constructor, "safe");
170        if (cached == null) {
171            cached = new SafeConstructor<>(constructor);
172            ReflectionCache.CONSTRUCT.put(constructor, cached, "safe");
173        }
174
175        return cached;
176    }
177}
