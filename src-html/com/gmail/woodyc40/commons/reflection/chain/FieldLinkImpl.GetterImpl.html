017package com.gmail.woodyc40.commons.reflection.chain;
018
019import com.gmail.woodyc40.commons.reflection.FieldManager;
020import com.gmail.woodyc40.commons.reflection.ReflectionTool;
021import com.gmail.woodyc40.commons.reflection.impl.ReflectAccess;
022import lombok.RequiredArgsConstructor;
023
024import java.lang.reflect.Field;
025import java.util.ArrayList;
026import java.util.List;
027
028/**
029 * Implementation of {@link com.gmail.woodyc40.commons.reflection.chain.FieldLink}. See description for the implemented
030 * interfaces.
031 *
032 * @author AgentTroll
033 * @version 1.0
034 * @since 1.0
035 */
036@RequiredArgsConstructor
037public class FieldLinkImpl implements FieldLink {
038    private final Class<?>                     base;
039    private final ReflectionChain              parent;
040    private       FieldManager<Object, Object> field;
041    private       Object                       instance;
042
043    @Override public FieldLink last(int index) {
044        this.instance = this.parent.returned.get(index);
045        return this;
046    }
047
048    @Override public FieldLink instance(Object instance) {
049        this.instance = instance;
050        return this;
051    }
052
053    @Override public FieldLink.Getter getter() {
054        return new GetterImpl();
055    }
056
057    @Override public FieldLink.Setter setter() {
058        return new SetterImpl();
059    }
060
061    @Override public FieldLink field(String name) {
062        this.field = ReflectAccess.accessField(ReflectionTool.forField(name, this.base));
063        return this;
064    }
065
066    @Override public FieldLink fieldFuzzy(Class<?> type, int index) {
067        List<Field> list = new ArrayList<>();
068        for (Field field : this.base.getDeclaredFields())
069            if (field.getType().equals(type))
070                list.add(field);
071        this.field = ReflectAccess.accessField(list.get(index));
072
073        return this;
074    }
075
076    @Override public FieldManager<?, ?> getManager() {
077        return this.field;
078    }
079
080    private class GetterImpl implements FieldLink.Getter {
081        @Override public ReflectionChain get() {
082            FieldLinkImpl.this.parent.returned.add(FieldLinkImpl.this.field.get(FieldLinkImpl.this.instance));
083            return FieldLinkImpl.this.parent;
084        }
085    }
086
087    private class SetterImpl implements FieldLink.Setter {
088        @Override public ReflectionChain set(Object value) {
089            FieldLinkImpl.this.field.set(FieldLinkImpl.this.instance, value);
090            return FieldLinkImpl.this.parent;
091        }
092    }
093}
