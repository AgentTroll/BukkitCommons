017package com.gmail.woodyc40.commons;
018
019import com.gmail.woodyc40.commons.concurrent.JavaFork;
020import com.gmail.woodyc40.commons.concurrent.ThreadPoolManager;
021import com.gmail.woodyc40.commons.event.Events;
022import com.gmail.woodyc40.commons.io.Files;
023import com.gmail.woodyc40.commons.nmsobc.protocol.Protocol;
024import com.gmail.woodyc40.commons.reflection.impl.ReflectionCache;
025import lombok.Getter;
026import org.bukkit.plugin.Plugin;
027import org.bukkit.plugin.java.JavaPlugin;
028
029import java.io.IOException;
030
031/**
032 * {@link org.bukkit.plugin.java.JavaPlugin} {@code class} representing this plugin utility
033 *
034 * @author AgentTroll
035 * @version 1.0
036 * @since 1.0
037 */
038public class Commons extends JavaPlugin {
039    /** The plugin instance */
040    @Getter private static Plugin plugin;
041
042    /**
043     * Gets the calling package for the method
044     *
045     * @param ignore {@code true} to prevent checks on BukkitCommons
046     * @return the package the called the method
047     */
048    public static Package getCaller(boolean ignore) {
049        StackTraceElement[] elements = Thread.currentThread().getStackTrace();
050        for (StackTraceElement element : elements) {
051            if (!element.getClassName().startsWith("java.") &&
052                (!ignore || !element.getClassName().startsWith("com.gmail.woodyc40.commons."))) {
053                Class<?> clazz = ReflectionCache.getClass(element.getClassName());
054                if (clazz.isAssignableFrom(JavaPlugin.class) || clazz.getPackage().getName()
055                                                                     .contains("com.gmail.woodyc40.commons"))
056                    return clazz.getPackage();
057            }
058        }
059
060        return null;
061    }
062
063    @Override public void onEnable() {
064        Commons.plugin = this;
065
066        try {
067            Files.update();
068            // JavaFork.start(); TODO
069        } catch (IOException x) {
070            x.printStackTrace();
071        }
072        Protocol.initiate(this);
073    }
074
075    @Override public void onDisable() {
076        ThreadPoolManager.shutdown();
077        JavaFork.shutdown();
078        Events.shutdown();
079    }
080}
