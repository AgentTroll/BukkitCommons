036/**
037 * {@link org.bukkit.plugin.java.JavaPlugin} {@code class} representing this plugin utility
038 *
039 * @author AgentTroll
040 * @version 1.0
041 * @since 1.0
042 */
043public class Commons extends JavaPlugin {
044    /** Classes that may be disabled due to availability */
045    private static final Class<?>[] CLASSES        = {
046            Instrument.class, UnsafeProvider.class, ReflectAccess.class,
047            ReflectionCache.getClass("com.gmail.woodyc40.commons.reflection.impl.ConstructorImpl"),
048            ReflectionCache.getClass("com.gmail.woodyc40.commons.reflection.impl.MethodImpl"),
049            ReflectionCache.getClass("com.gmail.woodyc40.commons.reflection.impl.FieldImpl")
050    };
051    /** Classes that need to be available for specific libraries */
052    private static final Class<?>[] UNSAFE_CLASSES = {
053            ReflectionCache.getClass("sun.misc.Unsafe"),
054            ReflectionCache.getClass("sun.reflect.MethodAccessor"),
055            ReflectionCache.getClass("sun.reflect.ConstantPool"),
056            ReflectionCache.getClass("sun.reflect.ConstructorAccessor"),
057            ReflectionCache.getClass("sun.reflect.ReflectionFactory"),
058            ReflectionCache.getClass("sun.misc.SharedSecrets")
059    };
060    /** The plugin instance */
061    @Getter private static Plugin plugin;
062
063    /**
064     * Gets the calling package for the method
065     *
066     * @param ignore {@code true} to prevent checks on BukkitCommons
067     * @return the package the called the method
068     */
069    public static Package getCaller(boolean ignore) {
070        StackTraceElement[] elements = Thread.currentThread().getStackTrace();
071        for (StackTraceElement element : elements) {
072            if (!element.getClassName().startsWith("java.") &&
073                (!ignore || !element.getClassName().startsWith("com.gmail.woodyc40.commons.")))
074                return ReflectionCache.getClass(element.getClassName()).getPackage();
075        }
076
077        return null;
078    }
079
080    /**
081     * Prevents the class loader from checking the commons jar of Google Guava
082     */
083    private static void disableClassLoaderSealing() {
084        Settings.setSafeReflection(true);
085        FieldManager<Package, URL> sealBase =
086                ReflectAccess.accessField(ReflectionTool.forField("sealBase", Package.class));
087
088        sealBase.set(Package.getPackage("com.google.commons"), null);
089        Settings.setSafeReflection(false);
090    }
091
092    @Override public void onEnable() {
093        Commons.plugin = this;
094        Commons.disableClassLoaderSealing();
095
096        for (Class<?> c : Commons.UNSAFE_CLASSES) {
097            if (c == null) {
098            }
099            // TODO disable CLASSES
100        }
101
102        Protocol.initiate(this);
103        try {
104            Files.update();
105            // JavaFork.start(); TODO
106        } catch (IOException x) {
107            x.printStackTrace();
108        }
109    }
110
111    @Override public void onDisable() {
112        ThreadPoolManager.shutdown();
113        JavaFork.shutdown();
114    }
115}
