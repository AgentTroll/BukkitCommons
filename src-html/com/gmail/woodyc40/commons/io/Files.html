017package com.gmail.woodyc40.commons.io;
018
019import com.gmail.woodyc40.commons.Commons;
020import lombok.Cleanup;
021import org.bukkit.configuration.file.FileConfiguration;
022import org.bukkit.configuration.file.YamlConfiguration;
023
024import java.io.File;
025import java.io.IOException;
026import java.util.Enumeration;
027import java.util.jar.JarEntry;
028import java.util.jar.JarFile;
029
030/**
031 * The related directories to which this plugin is required to know
032 *
033 * @author AgentTroll
034 * @version 1.0
035 * @since 1.0
036 */
037public final class Files {
038    /** Cached plugin directory */
039    private static File plugins;
040    /** Cached jar file */
041    private static File jar;
042    /** Cached java executable */
043    private static File executable;
044
045    private Files() {}
046
047    /**
048     * Get the cached plugin directory
049     *
050     * @return the plugin directory
051     */
052    public static File pluginDir() {
053        return Files.plugins;
054    }
055
056    /**
057     * Get the cached jar file representing the compressed classes of this library
058     *
059     * @return the jar file
060     */
061    public static File jarFile() {
062        return Files.jar;
063    }
064
065    /**
066     * Get the cached java executable
067     *
068     * @return the executable currently running the program
069     */
070    public static File executable() {
071        return Files.executable;
072    }
073
074    /**
075     * Updates the caches
076     *
077     * @throws IOException when the jar file cannot be accessed
078     */
079    public static void update() throws IOException {
080        File dir = new File(Commons.class.getProtectionDomain()
081                                         .getCodeSource()
082                                         .getLocation()
083                                         .getPath()
084                                         .replaceAll("%20", " "));
085        Files.plugins = new File(dir.getParentFile().getPath());
086
087        for (File file : Files.plugins.listFiles()) {
088            if (file.isDirectory())
089                continue;
090
091            @Cleanup JarFile jarFile = new JarFile(file);
092            Enumeration<JarEntry> entries = jarFile.entries();
093            while (entries.hasMoreElements()) {
094                JarEntry entry = entries.nextElement();
095                if (!"plugin.yml".equals(entry.getName()))
096                    continue;
097
098                // Deprecation does not make a difference in this case - it will still work regardless
099                FileConfiguration config = YamlConfiguration.loadConfiguration(jarFile.getInputStream(entry));
100                if (config.get("main").equals(Commons.class.getName()))
101                    Files.jar = file;
102            }
103        }
104
105        String executable = "java";
106        if (System.getProperty("os.name").toLowerCase().contains("win"))
107            executable += ".exe";
108        Files.executable = new File(new File(System.getProperty("java.home"), "bin"), executable);
109    }
110}
