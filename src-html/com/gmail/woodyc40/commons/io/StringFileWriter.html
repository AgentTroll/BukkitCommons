017package com.gmail.woodyc40.commons.io;
018
019import java.io.*;
020
021/**
022 * Allows for the functionality to write content to a file
023 * <p>
024 * <p>Must be closed when finished!</p>
025 *
026 * @author AgentTroll
027 * @version 1.0
028 * @since 1.0
029 */
030public class StringFileWriter implements Closeable {
031    /** The file to write to */
032    private final File           file;
033    /** The file writer */
034    private final BufferedWriter writer;
035
036    /**
037     * Builds a new file writer capable of writing Strings to the file
038     *
039     * @param file the file to write to
040     * @throws IOException if the BufferedWriter cannot be created
041     */
042    public StringFileWriter(File file) throws IOException {
043        this.file = file;
044        this.writer = new BufferedWriter(new FileWriter(file));
045    }
046
047    /**
048     * Writes a line to the file
049     *
050     * @param line the String to write
051     */
052    public void writeLine(String line) {
053        try {
054            this.writer.write(line + '\n');
055        } catch (IOException e) {
056            e.printStackTrace();
057        }
058    }
059
060    /**
061     * Writes multiple lines to the file
062     *
063     * @param lines the lines to write
064     */
065    public void writeLine(String... lines) {
066        for (String line : lines) {
067            this.writeLine(line);
068        }
069    }
070
071    /**
072     * Adds text to the file. This does not append a new line.
073     *
074     * @param string the string to write
075     */
076    public void write(String string) {
077        try {
078            this.writer.write(string);
079        } catch (IOException e) {
080            e.printStackTrace();
081        }
082    }
083
084    /**
085     * Adds multiple pieces of text to the file. This does not append a new line for each string.
086     *
087     * @param strings the strings to write
088     */
089    public void write(String... strings) {
090        for (String string : strings) {
091            this.write(string);
092        }
093    }
094
095    @Override public void close() throws IOException {
096        this.writer.close();
097    }
098}
