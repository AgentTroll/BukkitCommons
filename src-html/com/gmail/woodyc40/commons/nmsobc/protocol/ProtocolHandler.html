020import com.gmail.woodyc40.commons.event.Events;
021import com.gmail.woodyc40.commons.nmsobc.CbServer;
022import com.gmail.woodyc40.commons.nmsobc.McServer;
023import com.gmail.woodyc40.commons.reflection.FieldManager;
024import com.gmail.woodyc40.commons.reflection.chain.ReflectionChain;
025import lombok.Getter;
026import lombok.RequiredArgsConstructor;
027import net.minecraft.util.io.netty.channel.*;
028import org.bukkit.Bukkit;
029import org.bukkit.entity.Player;
030import org.bukkit.event.EventHandler;
031import org.bukkit.event.Listener;
032import org.bukkit.event.player.*;
033import org.bukkit.event.server.PluginDisableEvent;
034import org.bukkit.plugin.Plugin;
035
036import java.util.HashMap;
037import java.util.Map;

039/**
040 * The packet handler that will listen for packet writes and reads, as well as listen for player join and leave to cache
041 * their connection to the server on behalf of the protocol package.
042 *
043 * @author AgentTroll
044 * @version 1.0
045 * @since 1.0
046 */
047public final class ProtocolHandler { // TODO server channels
048    /** The connection cache that holds netty channels */
049    @Getter private static final Map<Player, Channel> cache = new HashMap<>();
050    /** The name of the handlers */
051    private final                String               name  = "ProtocolHandler - BukkitCommons";
052    /** The plugin that is activated by this handler. Will be {@code null} when disabled */
053    private volatile Plugin plugin;
054
055    /**
056     * Inits this handler. Registers internal listeners. Sets the plugin flag.
057     *
058     * @param plugin the plugin to use in this handler.
059     */
060    public ProtocolHandler(Plugin plugin) {
061        Bukkit.getServer().getPluginManager()
062              .registerEvents(new ProtocolListener(), plugin);
063        this.plugin = plugin;
064    }
065
066    private void putProxy(Channel channel, Player player) {
067        channel.pipeline().addLast(this.name, new ProtocolHandler.PacketAdapter(player));
068    }
069
070    // Private classes are shielded from access.
071    // They are handled correctly internally.
072    // If not really "inaccessible",
073    // At least very difficult to get to,
074    // Slow (or fast) reflect operations must be performed
075    // On classlevel native search using forName(String)
076
077    /**
078     * Represents the handler inserted first into the minecraft protocol manager to handle incoming and outgoing
079     * connection packets, and fire the event in the event framework.
080     *
081     * @author AgentTroll
082     * @version 1.0
083     * @since 1.0
084     */
085    @RequiredArgsConstructor
086    private class PacketAdapter extends ChannelDuplexHandler {
087        /** The player that this intercepts. Can be null for server channels */
088        private final Player player;
089
090        /**
091         * Handles incoming packets
092         *
093         * @param ctx the context which the packet is received in
094         * @param msg the packet itself
095         * @throws Exception if an error occurs
096         */
097        @Override public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
098            CustomEvent event = new PacketEvent(msg, this.player, PacketEvent.Bound.SERVER_BOUND);
099
100            Events.call(event);
101            if (event.getCancelled()) return;
102
103            super.channelRead(ctx, msg);
104        }
105
106        /**
107         * Handles outgoing packets
108         *
109         * @param ctx     the context which the packet is sent in
110         * @param msg     the packet itself
111         * @param promise the promise made to the channel
112         * @throws Exception if an error occurs
113         */
114        @Override public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) throws Exception {
115            CustomEvent event = new PacketEvent(msg, this.player, PacketEvent.Bound.CLIENT_BOUND);
116
117            Events.call(event);
118            if (event.getCancelled()) return;
119
120            super.write(ctx, msg, promise);
121        }
122
123        /**
124         * Handles exceptions occurring in the Channel
125         *
126         * @param ctx   the context which the exception occurs in
127         * @param cause the cause of the exception
128         * @throws Exception if yet another exception occurs
129         */
130        @Override public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
131            cause.printStackTrace();
132        }
133    }
134
135    /**
136     * The internal listener that adds the handlers to the player's connection and cleans up after the handler
137     *
138     * @author AgentTroll
139     * @version 1.0
140     * @since 1.0
141     */
142    private class ProtocolListener implements Listener {
143        /** Represents the channel ("m" as of 1.7.10 snapshot) field in NetworkManager */
144        private final FieldManager<Object, Channel> CHANNEL =
145                (FieldManager<Object, Channel>) new ReflectionChain(McServer.getClass("NetworkManager"))
146                        .field().fieldFuzzy(Channel.class, 0).getManager();
147
148        /**
149         * When a player joins
150         *
151         * @param event the event called
152         */
153        @EventHandler public void join(PlayerJoinEvent event) {
154            if (ProtocolHandler.this.plugin == null) return; // Whoopsie, plugin disabled.
155
156            Channel conn = this.CHANNEL.get(
157                    new ReflectionChain(CbServer.cPlayerClass())
158                            .method().method("getHandle").param(event.getPlayer()).invoker().invoke()      // 0
159                            .field(McServer.getEPlayer()).field("playerConnection").last(0).getter().get() // 1
160                            .field().field("networkManager").getter().get()                                // 2
161                            .reflect());
162
163            ProtocolHandler.this.putProxy(conn, event.getPlayer());
164            ProtocolHandler.getCache().put(event.getPlayer(), conn);
165        }
166
167        /**
168         * When a player leaves
169         *
170         * @param event the event called
171         */
172        @EventHandler public void leave(PlayerQuitEvent event) {
173            this.extract(event.getPlayer());
174        } // In case some idiot leaves his Player objects lying around
175
176        /**
177         * When a player is kicked
178         *
179         * @param event the event called
180         */
181        @EventHandler public void kick(PlayerKickEvent event) {
182            this.extract(event.getPlayer());
183        } // Same with this
184
185        /**
186         * When the plugin is disabled
187         *
188         * @param event the event called
189         */
190        @EventHandler public void disable(PluginDisableEvent event) {
191            if (event.getPlugin().equals(ProtocolHandler.this.plugin)) {
192                ProtocolHandler.this.plugin = null; // Marker
193                for (Player player : Bukkit.getServer().getOnlinePlayers())
194                    this.extract(player);
195            }
196        }
197
198        /**
199         * Removes the handler from the player
200         *
201         * @param player the player to remove the handler from
202         */
203        public void extract(Player player) {
204            final Channel conn = ProtocolHandler.getCache().get(player);
205            if (conn == null) return;
206
207            conn.eventLoop().execute(new Runnable() {
208                @Override
209                public void run() {
210                    if (conn.pipeline().names().contains(ProtocolHandler.this.name))
211                        conn.pipeline().remove(ProtocolHandler.this.name);
212                }
213            });
214            ProtocolHandler.getCache().remove(player);
215        }
216    }
217}
