017package com.gmail.woodyc40.commons.providers;
018
019import lombok.Getter;
020import sun.misc.Unsafe;
021
022import java.lang.reflect.*;
023import java.util.HashMap;
024import java.util.Map;

026/**
027 * This provides a wrapper for {@link Unsafe}, which allows for low-level Java operations unsafely. <p> <p>Note that the
028 * starting method used to acquire the instance of Unsafe may throw two reflective exceptions caused by access
029 * restrictions to the instance field of Unsafe. The deeper cause of that may be the JVM used to run this. This is a
030 * HotSpot directed program.</p> <p> <p>This class is not thread safe</p>
031 *
032 * @author AgentTroll
033 * @version 1.0
034 * @see #getProvider()
035 * @see Unsafe#getUnsafe()
036 * @since 1.0
037 */
038public final class UnsafeProvider {
039    /** The Unsafe instance. */
040    private static final         Unsafe           PROVIDER = UnsafeProvider.initUnsafe();
041    /** Cached field offsets */
042    @Getter private static final Map<Field, Long> offsets  = new HashMap<>();
043
044    /**
045     * Should not be instantiated, hence the private constructor.
046     */
047    private UnsafeProvider() {}
048
049    /**
050     * Used to acquire the instance of {@link Unsafe}. <p> <p>This is more of a "hack" than a direct method, as it uses
051     * reflection in order to obtain the stored instance of the Unsafe singleton. Unsafe cannot be accessed through its
052     * singleton {@link Unsafe#getUnsafe()} method, as it checks the calling {@code class} to make sure that the {@link
053     * ClassLoader} is {@code null}. Java {@code class}es don't use a {@link ClassLoader}, hence the reason why they can
054     * call {@link Unsafe#getUnsafe()} without {@link SecurityException}.</p> <p> <p>Being a "hack", this method will
055     * never be guaranteed to actually get an instance of Unsafe, the field that stores the instance of Unsafe
056     * <strong>is named differently on different implementations of the JVM</strong>. This specific method is designed
057     * to get Unsafe from the Java HotSpot JVM.</p>
058     *
059     * @return the Unsafe instance acquired from the HotSpot JVM Unsafe {@code class}
060     */
061    private static Unsafe initUnsafe() {
062        try {
063            Field field = Unsafe.class.getDeclaredField("theUnsafe");
064            field.setAccessible(true);
065            return (Unsafe) field.get(null);
066        } catch (NoSuchFieldException | IllegalAccessException x) {
067            x.printStackTrace();
068        }
069        return null;
070    }
071
072    /**
073     * Acquires the raw instance Unsafe "hacked" from the source. <p> <p>If an exception was thrown when this class was
074     * loaded, this method will most likely {@code return null}. JVM implementations may modify the field in which the
075     * instance of Unsafe is stored, throwing an exception when normal acquisition methods are applied.</p> <p> <p>This
076     * is guaranteed to <strong>not</strong> {@code return null} on the HotSpot JVM</p>
077     *
078     * @return the Unsafe instance received when this {@code class} was loaded.
079     * @throws IllegalStateException if the Unsafe instance was not acquired already during load
080     */
081    public static Unsafe getProvider() {
082        if (UnsafeProvider.PROVIDER == null)
083            throw new IllegalStateException("Cannot be used, Unsafe is null");
084
085        return UnsafeProvider.PROVIDER;
086    }
087
088    // Reflection area
089
090    /**
091     * Util method to get whether or not a field is {@code static}.
092     *
093     * @param field the field to check for a {@code static} modifier
094     * @return {@code true} to indicate that a field is {@code static}, {@code false} if the field does not have a
095     * {@code static} modifier
096     */
097    private static boolean fieldStatic(Member field) {
098        return Modifier.isStatic(field.getModifiers());
099    }
100
101    /**
102     * Provides field offset calculations of a field in order to find its location in a {@code class}. <p> <p>The field
103     * parameter should never be {@code null}. An exception will be thrown if it is.</p>
104     *
105     * @param field the field to {@code return} the offset of
106     * @return the relative {@code class} location offset of the field
107     */
108    public static long fieldOffset0(Field field) {
109        if (UnsafeProvider.fieldStatic(field))
110            return UnsafeProvider.PROVIDER.staticFieldOffset(field);
111        return UnsafeProvider.PROVIDER.objectFieldOffset(field);
112    }
113
114    /**
115     * Obtains the cached offset for the given field. This is almost 4 times faster than calculating each time at the
116     * cost of memory and thread safety.
117     *
118     * @param field the field to {@code return} the offset of
119     * @return the field offset
120     */
121    public static long fieldOffset(Field field) { // TODO concurrency
122        Long offset = UnsafeProvider.offsets.get(field);
123        if (offset == null) {
124            offset = UnsafeProvider.fieldOffset0(field);
125            UnsafeProvider.offsets.put(field, offset);
126        }
127
128        return offset;
129    }
130
131    /**
132     * Acquires the raw field value of the specified field of a {@code class}. <p> <p>It may {@code return null}. This
133     * can happen under the following circumstances: <ul> <li>If the field value is {@code null} for the particular
134     * instance (More common).</li> <li>If one of the parameters required to be non-{@code null} is {@code null}</li>
135     * <li>Or if the field is not found or not able to be accessed. Check method call, spelling, and class source.
136     * Indicated by a thrown exception.</li> </ul> </p> <p> <p>The holder parameter may be null for static fields. In
137     * fact, it is set null for this purpose in the method itself, so it will not matter nevertheless.</p>
138     *
139     * @param holder the instance of the object to acquire the value from.
140     * @return the value of the acquired field from the instance of the holder parameter
141     * @throws IllegalArgumentException if a nullable parameter is {@code null}
142     */
143    public static Object acquireField(Field field, Object holder) {
144        long offset = UnsafeProvider.fieldOffset(field);
145        if (UnsafeProvider.fieldStatic(field))
146            return UnsafeProvider.PROVIDER.getObject(field.getDeclaringClass(), offset);
147
148        return UnsafeProvider.PROVIDER.getObject(holder, offset);
149    }
150
151    /**
152     * Sets the raw field value for the given field object in the class <p> <p>This method will fail under the following
153     * circumstances: <ul> <li>If one of the parameters required to be non-{@code null} is {@code null}</li> <li>Or the
154     * field could not be found. Indicated by a thrown exception</li> </ul> </p>
155     *
156     * @param field  the field to put the new value in
157     * @param holder the instance of the object to set the field in
158     * @param value  the new value of the field
159     */
160    public static void setField(Field field, Object holder, Object value) {
161        long offset = UnsafeProvider.fieldOffset(field);
162
163        if (UnsafeProvider.fieldStatic(field)) {
164            UnsafeProvider.PROVIDER.putObject(field.getDeclaringClass(), offset, value);
165            return;
166        }
167
168        UnsafeProvider.PROVIDER.putObject(holder, offset, value);
169    }
170
171    /**
172     * Instantiates a {@code class} without calling the constructor of the {@code class} <p> <p><strong>This is
173     * EXTREMELY IMPORTANT - THIS METHOD WILL <u>NOT</u> CALL THE {@code CLASS} CONSTRUCTOR!</strong></p>
174     *
175     * @param <T> the type of {@code class} instantiated, the type returned
176     * @param c   the {@code class} to make a new instance of
177     * @return the new instance of the {@code class} passed in
178     */
179    public static <T> T initClass(Class<T> c) {
180        try {
181            return (T) UnsafeProvider.PROVIDER.allocateInstance(c);
182        } catch (InstantiationException x) {
183            x.printStackTrace();
184        }
185        return null;
186    }

188    // Concurrency area
189
190    /**
191     * Forces current thread to acquire or release the monitor of an object. <p> <p>This is a very low level
192     * implementation, no checks will be performed to whether the monitor is already held by the current thread before
193     * execution.</p> <p> <p>It is <strong>crucial</strong> to use this as an implementation or entry point to
194     * synchronization instead of a careless direct call. A monitor acquired with this method can only be released
195     * using: {@code UnsafeProvider.monitor(false, object);} where {@code object} is the object with its monitor held by
196     * the current thread.</p>
197     *
198     * @param entry {@code true} to acquire the monitor, {@code false} to release.
199     * @param o     The object to acquire to release the monitor of.
200     */
201    public static void monitor(boolean entry, Object o) {
202        if (entry) {
203            UnsafeProvider.PROVIDER.monitorEnter(o);
204            return;
205        }
206        UnsafeProvider.PROVIDER.monitorExit(o);
207    }
208
209    /**
210     * CAS operation, sets the field in an object to nw if expect is the current value. <p> <p>Note that this method
211     * performs a deep search for the field to swap.</p>
212     *
213     * @param instance the object instance to acquire the field in
214     * @param field    the field to swap the value for
215     * @param expect   the expected value of the field
216     * @param nw       the value to set if expect is the current field value
217     */
218    public static void compareAndSwap(Object instance, Field field, Object expect, Object nw) {
219        while (!UnsafeProvider.PROVIDER.compareAndSwapObject(instance, UnsafeProvider.fieldOffset(field), expect, nw))
220            UnsafeProvider.setField(field, instance, nw);
221    }

225    /**
226     * Takes the shallow size of the object. <p> <p>Equivalent Java implementation of C sizeOf</p> <p> <p>Is not an
227     * alternative for {@link java.lang.instrument.Instrumentation#getObjectSize(Object)}. Will not be as accurate as
228     * that method, this should only be used for programs where the EXACT size of an object in heap can have small
229     * inaccuracies.</p>
230     *
231     * @param object the object to find the size of. Cannot be an array.
232     * @return the size of the object stored in its header in the JVM
233     */
234    public static long sizeOf(Object object) {
235        return UnsafeProvider.PROVIDER.getAddress(
236                UnsafeProvider.PROVIDER.getLong(object, 4L) + 12L);
237    }
238
239    /**
240     * Makes an unsafe cast from main to a superclass without {@link java.lang.ClassCastException}.
241     *
242     * @param main       the {@code class} to cast
243     * @param superclass the {@code class} to cast main to
244     * @param <T>        the type of superclass to cast to
245     * @return the result of casting main to superclass
246     */
247    public static <T> T castSuper(Object main, T superclass) {
248        UnsafeProvider.PROVIDER.putAddress(UnsafeProvider.normalize(UnsafeProvider.PROVIDER.getInt(main, 4L)) + 36L,
249                                           UnsafeProvider.normalize(UnsafeProvider.PROVIDER.getInt(superclass, 4L)));
250
251        return (T) main;
252    }
253
254    /**
255     * Converts a signed {@code int} to an unsigned {@code long}.
256     *
257     * @param value the {@code int} to convert
258     * @return the derived unsigned {@code long} associated with the value
259     */
260    public static long normalize(int value) {
261        if (value >= 0)
262            return (long) value;
263        return ~0L >>> 32 & (long) value;
264    }
265
266    /**
267     * Converts a signed {@code long} to unsigned {@code long}.
268     *
269     * @param value the {@code long} to convert
270     * @return the derived unsigned {@code long} associated with the value
271     * @deprecated longs can hold unsigned values natively
272     */
273    @Deprecated public static long normalize(long value) {
274        if (value >= 0L)
275            return value;
276        return ~0L >>> 32 & value;
277    }

279    /**
280     * Throws an unchecked exception, meaning the exception should not be caught
281     *
282     * @param x the exception to throw unchecked
283     */
284    public static void exception(Exception x) {
285        UnsafeProvider.PROVIDER.throwException(x);
286    }
