017package com.gmail.woodyc40.commons.concurrent.collect;
018
019import com.gmail.woodyc40.commons.providers.UnsafeProvider;
020import com.gmail.woodyc40.commons.reflection.ReflectionTool;
021import lombok.Getter;
022import lombok.RequiredArgsConstructor;
023import sun.misc.Unsafe;
024
025import javax.annotation.concurrent.ThreadSafe;
026
027/**
028 * Implementation of the internal lock structure. Uses the {@link java.util.concurrent.locks.LockSupport} algorithm
029 * for locking and unlocking.
030 *
031 * @author AgentTroll
032 * @version 1.0
033 * @since 1.1
034 */
035@ThreadSafe
036public class UnsafeLock implements InternalLock {
037    /** Provided unsafe access to increase the throughput of concurrent ops */
038    private static final Unsafe UNSAFE   = UnsafeProvider.getProvider();
039    /** Lock state for "held by other thread" */
040    private static final int    LOCKED   = 1;
041    /** Lock state for "free to acquire" */
042    private static final int    UNLOCKED = 0;
043
044    static {
045        UnsafeLock.stateOff = UnsafeLock.UNSAFE.objectFieldOffset(ReflectionTool.forField("state", UnsafeLock.class));
046        UnsafeLock.tailOff = UnsafeLock.UNSAFE.objectFieldOffset(ReflectionTool.forField("tail", UnsafeLock.class));
047        UnsafeLock.headOff = UnsafeLock.UNSAFE.objectFieldOffset(ReflectionTool.forField("head", UnsafeLock.class));
048
049        UnsafeLock.nextOff = UnsafeLock.UNSAFE.objectFieldOffset(ReflectionTool.forField("next",
050                                                                                         UnsafeLock.ThreadNode.class));
051    }
052
053    /** The tail offset */
054    private static long tailOff;
055    /** The head offset */
056    private static long headOff;
057    /** The head offset */
058    private static long stateOff;
059
060    /** The next node offset in ThreadNode */
061    private static long nextOff;
062
063    /** Thread node linked list head */
064    private volatile UnsafeLock.ThreadNode head = new UnsafeLock.ThreadNode(null); // Do not listen to "can make final"
065    /** Thread node linked list tail */
066    private volatile UnsafeLock.ThreadNode tail = this.head;
067    /** The lock state */
068    private volatile int state;
069
070    @Override public void lock() {
071        Thread thread = Thread.currentThread();
072        this.insert(thread);
073        boolean interrupt = false;
074
075        while (this.read().getThread() != thread || !this.updateState(UnsafeLock.UNLOCKED, UnsafeLock.LOCKED)) {
076            UnsafeLock.UNSAFE.park(false, 0L);
077            if (Thread.interrupted())
078                interrupt = true;
079        }
080
081        this.readAndRemove();
082        if (interrupt)
083            thread.interrupt();
084    }
085
086    @Override public void unlock() {
087        this.state = UnsafeLock.UNLOCKED;
088        UnsafeLock.ThreadNode node = this.read();
089        if (node == null) return;
090        UnsafeLock.UNSAFE.unpark(node.getThread());
091    }
092
093    /**
094     * Performs a CAS operation on the state
095     *
096     * @param expected the expected state
097     * @param state    the new state to be set
098     * @return {@code true} if the operation succeeds
099     */
100    private boolean updateState(int expected, int state) {
101        return UnsafeLock.UNSAFE.compareAndSwapInt(this, UnsafeLock.stateOff, expected, state);
102    }
103
104    /**
105     * Performs CAS operation to replace the tail
106     *
107     * @param expected the expected tail
108     * @param tail     the new tail to set
109     */
110    private boolean updateTail(UnsafeLock.ThreadNode expected, UnsafeLock.ThreadNode tail) {
111        return UnsafeLock.UNSAFE.compareAndSwapObject(this, UnsafeLock.tailOff, expected, tail);
112    }
113
114    /**
115     * Performs CAS operation to replace the head
116     *
117     * @param expected the expected head
118     * @param next the next node to set
119     */
120    private boolean updateHead(UnsafeLock.ThreadNode expected, UnsafeLock.ThreadNode next) {
121        return UnsafeLock.UNSAFE.compareAndSwapObject(this, UnsafeLock.headOff, expected, next);
122    }
123
124    /**
125     * Performs CAS operation to replace the next node in the thread node
126     *
127     * @param inst     the instance of ThreadNode to set the next
128     * @param expected the expected next node
129     * @param next     the next node to set
130     */
131    private boolean updateNext(UnsafeLock.ThreadNode inst, UnsafeLock.ThreadNode expected, UnsafeLock.ThreadNode next) {
132        return UnsafeLock.UNSAFE.compareAndSwapObject(inst, UnsafeLock.nextOff, expected, next);
133    }
134
135    /**
136     * Inserts the node into the end of the linked list using the Michael & Scott algorithm found on
137     * <a href="http://goo.gl/oUaQUT">DevWorks</a>
138     *
139     * @param thread the thread to add to the linked list
140     */
141    private void insert(Thread thread) {
142        UnsafeLock.ThreadNode n = new UnsafeLock.ThreadNode(thread);
143        for (; ; ) {
144            UnsafeLock.ThreadNode t = this.tail;
145            UnsafeLock.ThreadNode s = t.getNext();
146            if (t == this.tail) {
147                if (s == null) {
148                    if (this.updateNext(t, null, n)) {
149                        this.updateTail(t, n);
150                        return;
151                    }
152                } else {
153                    this.updateTail(t, s);
154                }
155            }
156        }
157    }
158
159    /**
160     * Reads the head of the linked list, remove it, and return the original head. Based off of the algorithm found
161     * <a href="http://goo.gl/oUaQUT">here</a>
162     *
163     * @return the original head of the linked list, before replacement with the next
164     */
165    private UnsafeLock.ThreadNode readAndRemove() {
166        for (; ; ) {
167            UnsafeLock.ThreadNode h = this.head;
168            UnsafeLock.ThreadNode t = this.tail;
169            UnsafeLock.ThreadNode first = h.getNext();
170            if (h == this.head) {
171                if (h == t) {
172                    if (first == null)
173                        return null;
174                    else
175                        this.updateTail(t, first);
176                } else if (this.updateHead(h, first)) {
177                    return first;
178                }
179            }
180        }
181    }
182
183    /**
184     * Reads the head of the linked list without removing the node, based off the algorithm found <a
185     * href="http://goo.gl/oUaQUT">here</a>
186     *
187     * @return the head of the linked list
188     */
189    private UnsafeLock.ThreadNode read() {
190        for (; ; ) {
191            UnsafeLock.ThreadNode h = this.head;
192            UnsafeLock.ThreadNode t = this.tail;
193            UnsafeLock.ThreadNode first = h.getNext();
194            if (h == this.head) {
195                if (h == t) {
196                    if (first == null) return null;
197                    else this.updateTail(t, first);
198                } else {
199                    if (first != null) return first;
200                    else this.updateHead(h, null);
201                }
202            }
203        }
204    }
205
206    /**
207     * Linked-list node part of the waiting list of threads acquiring the lock
208     *
209     * @author AgentTroll
210     * @version 1.0
211     * @since 1.1
212     */
213    @RequiredArgsConstructor
214    private class ThreadNode {
215        /** The thread held by the node */
216        @Getter private final    Thread                thread;
217        /** The next node in the linked list */
218        @Getter private volatile UnsafeLock.ThreadNode next;
219    }
