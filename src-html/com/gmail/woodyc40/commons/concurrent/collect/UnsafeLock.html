017package com.gmail.woodyc40.commons.concurrent.collect;
018
019import com.gmail.woodyc40.commons.providers.UnsafeProvider;
020import com.gmail.woodyc40.commons.reflection.ReflectionTool;
021import lombok.Getter;
022import lombok.RequiredArgsConstructor;
023import sun.misc.Unsafe;
024
025/**
026 * Implementation of the internal lock structure
027 *
028 * @author AgentTroll
029 * @version 1.0
030 * @since 1.1
031 */
032public class UnsafeLock implements InternalLock {
033    /** Provided unsafe access to increase the throughput of concurrent ops */
034    private static final Unsafe UNSAFE   = UnsafeProvider.getProvider();
035    /** Lock state for "held by other thread" */
036    private static final int    LOCKED   = 1;
037    /** Lock state for "free to acquire" */
038    private static final int    UNLOCKED = 0;
039
040    static {
041        UnsafeLock.stateOff = UnsafeLock.UNSAFE.objectFieldOffset(ReflectionTool.forField("state", UnsafeLock.class));
042        UnsafeLock.tailOff = UnsafeLock.UNSAFE.objectFieldOffset(ReflectionTool.forField("tail", UnsafeLock.class));
043        UnsafeLock.headOff = UnsafeLock.UNSAFE.objectFieldOffset(ReflectionTool.forField("head", UnsafeLock.class));
044        UnsafeLock.currentOff = UnsafeLock.UNSAFE.objectFieldOffset(ReflectionTool.forField("current",
045                                                                                            UnsafeLock.class));
046
047        UnsafeLock.nextOff = UnsafeLock.UNSAFE.objectFieldOffset(ReflectionTool.forField("next",
048                                                                                         UnsafeLock.ThreadNode.class));
049    }
050
051    /** The tail offset */
052    private static long tailOff;
053    /** The head offset */
054    private static long headOff;
055    /** The head offset */
056    private static long stateOff;
057    /** The current offset */
058    private static long currentOff;
059
060    /** The next node offset in ThreadNode */
061    private static long nextOff;
062
063    /** Thread node linked list head */
064    private volatile UnsafeLock.ThreadNode head = new UnsafeLock.ThreadNode(null); // Do not listen to "can make final"
065    /** Thread node linked list tail */
066    private volatile UnsafeLock.ThreadNode tail = this.head;
067    /** The lock state */
068    private volatile int state;
069    /** The current holder of the lock */
070    private volatile UnsafeLock.ThreadNode current = new UnsafeLock.ThreadNode(null);
071
072    @Override public void lock() {
073        Thread thread = Thread.currentThread();
074        boolean interrupt = false;
075
076        while (this.state == UnsafeLock.LOCKED && thread != this.current.getThread()) {
077            this.insert(new UnsafeLock.ThreadNode(thread));
078
079            if (Thread.interrupted())
080                interrupt = true;
081            UnsafeLock.UNSAFE.park(false, 0L);
082        }
083
084        if (this.updateState(UnsafeLock.UNLOCKED, UnsafeLock.LOCKED))
085            this.updateCurrent(this.current, new UnsafeLock.ThreadNode(thread));
086
087        if (interrupt)
088            thread.interrupt();
089    }
090
091    @Override public void unlock() {
092        if (this.updateState(UnsafeLock.LOCKED, UnsafeLock.UNLOCKED)) {
093            UnsafeLock.ThreadNode head = this.readAndRemove();
094            if (head == null) return;
095            if (head.getThread() == null) return;
096
097            UnsafeLock.UNSAFE.unpark(head.getThread());
098        }
099    }
100
101    /**
102     * Performs a CAS operation on the state
103     *
104     * @param expected the expected state
105     * @param state    the new state to be set
106     * @return {@code true} if the operation succeeds
107     */
108    private boolean updateState(int expected, int state) {
109        return UnsafeLock.UNSAFE.compareAndSwapInt(this, UnsafeLock.stateOff, expected, state);
110    }
111
112    /**
113     * Performs a CAS operation on the current thread
114     *
115     * @param expected the expected current node
116     * @param current  the new state to be set
117     * @return {@code true} if the operation succeeds
118     */
119    private boolean updateCurrent(UnsafeLock.ThreadNode expected, UnsafeLock.ThreadNode current) {
120        return UnsafeLock.UNSAFE.compareAndSwapObject(this, UnsafeLock.currentOff, expected, current);
121    }
122
123    /**
124     * Performs CAS operation to replace the tail
125     *
126     * @param expected the expected tail
127     * @param tail     the new tail to set
128     */
129    private boolean updateTail(UnsafeLock.ThreadNode expected, UnsafeLock.ThreadNode tail) {
130        return UnsafeLock.UNSAFE.compareAndSwapObject(this, UnsafeLock.tailOff, expected, tail);
131    }
132
133    /**
134     * Performs CAS operation to replace the head
135     *
136     * @param expected the expected head
137     * @param next the next node to set
138     */
139    private boolean updateHead(UnsafeLock.ThreadNode expected, UnsafeLock.ThreadNode next) {
140        return UnsafeLock.UNSAFE.compareAndSwapObject(this, UnsafeLock.headOff, expected, next);
141    }
142
143    /**
144     * Performs CAS operation to replace the next node in the thread node
145     *
146     * @param inst     the instance of ThreadNode to set the next
147     * @param expected the expected next node
148     * @param next     the next node to set
149     */
150    private boolean updateNext(UnsafeLock.ThreadNode inst, UnsafeLock.ThreadNode expected, UnsafeLock.ThreadNode next) {
151        return UnsafeLock.UNSAFE.compareAndSwapObject(inst, UnsafeLock.nextOff, expected, next);
152    }
153
154    /**
155     * Inserts the node into the end of the linked list using the Michael & Scott algorithm found on
156     * <a href="http://www.ibm.com/developerworks/library/j-jtp04186/">DevWorks</a>
157     *
158     * @param node the node to insert
159     */
160    private void insert(UnsafeLock.ThreadNode node) {
161        while (true) {
162            UnsafeLock.ThreadNode curTail = this.tail;
163            UnsafeLock.ThreadNode residue = curTail.getNext();
164
165            if (curTail == this.tail) {
166                if (residue == null) {
167                    if (this.updateNext(curTail, null, node)) {
168                        this.updateTail(curTail, node);
169                        return;
170                    }
171                } else this.updateTail(curTail, residue);
172            }
173        }
174    }
175
176    /**
177     * Reads the head of the linked list, remove it, and return the original head. Based off of the algorithm found <a
178     * href="http://fuseyism.com/classpath/doc/java/util/concurrent/ConcurrentLinkedQueue-source.html">here</a>
179     *
180     * @return the original head of the linked list, before replacement with the next
181     */
182    private UnsafeLock.ThreadNode readAndRemove() {
183        while (true) {
184            UnsafeLock.ThreadNode curHead = this.head;
185            UnsafeLock.ThreadNode curTail = this.tail;
186            UnsafeLock.ThreadNode residue = curHead.getNext();
187
188            if (curHead == this.head) {
189                if (curHead == curTail) {
190                    if (residue == null) return null;
191                    else this.updateTail(curTail, residue);
192                } else if (this.updateHead(curHead, residue))
193                    return residue;
194            }
195        }
196    }
197
198    /**
199     * Linked-list node part of the waiting list of threads acquiring the lock
200     *
201     * @author AgentTroll
202     * @version 1.0
203     * @since 1.1
204     */
205    @RequiredArgsConstructor
206    private class ThreadNode {
207        /** The thread held by the node */
208        @Getter private final    Thread                thread;
209        /** The next node in the linked list */
210        @Getter private volatile UnsafeLock.ThreadNode next;
211    }
212}
