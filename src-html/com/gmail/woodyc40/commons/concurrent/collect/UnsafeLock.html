017package com.gmail.woodyc40.commons.concurrent.collect;
018
019import com.gmail.woodyc40.commons.providers.UnsafeProvider;
020import com.gmail.woodyc40.commons.reflection.ReflectionTool;
021import lombok.Getter;
022import sun.misc.Unsafe;
023
024/**
025 * Implementation of the internal lock structure
026 *
027 * @author AgentTroll
028 * @version 1.0
029 * @since 1.1
030 */
031public class UnsafeLock implements InternalLock {
032    /** Provided unsafe access to increase the throughput of concurrent ops */
033    private static final Unsafe UNSAFE   = UnsafeProvider.getProvider();
034    /** Lock state for "held by other thread" */
035    private static final int    LOCKED   = 1;
036    /** Lock state for "free to acquire" */
037    private static final int    UNLOCKED = 0;
038
039    static {
040        UnsafeLock.stateOff = UnsafeProvider.fieldOffset0(ReflectionTool.forField("state", UnsafeLock.class));
041        UnsafeLock.tailOff = UnsafeProvider.fieldOffset0(ReflectionTool.forField("tail", UnsafeLock.class));
042    }
043
044    /** The tail offset */
045    private static long tailOff;
046    /** Offset for the lock state, used for atomic updates */
047    private static long stateOff;
048
049    /** Thread node linked list tail */
050    private volatile UnsafeLock.ThreadNode tail = new UnsafeLock.ThreadNode(null);
051    /** The lock state */
052    private volatile int state;
053
054    @Override public void lock() {
055        while (!this.updateState(UnsafeLock.UNLOCKED, UnsafeLock.LOCKED)) {
056            Thread thread = Thread.currentThread();
057            if (!thread.equals(this.tail.getThread())) {
058                this.updateTail(this.tail, new UnsafeLock.ThreadNode(thread));
059                UnsafeLock.UNSAFE.park(false, 0L);
060            }
061        }
062
063        this.updateTail(this.tail, new UnsafeLock.ThreadNode(Thread.currentThread()));
064    }
065
066    @Override public void unlock() {
067        if (this.updateState(UnsafeLock.LOCKED, UnsafeLock.UNLOCKED)) {
068            UnsafeLock.ThreadNode previous = this.tail.getPrevious();
069            if (previous == null) return;
070
071            this.updateTail(this.tail, previous);
072            UnsafeLock.UNSAFE.unpark(this.tail.getThread());
073        }
074    }
075
076    /**
077     * Performs a CAS operation on the state
078     *
079     * @param expected the expected state
080     * @param state    the new state to be set
081     * @return {@code true} if the operation succeeds
082     */
083    private boolean updateState(int expected, int state) {
084        return UnsafeLock.UNSAFE.compareAndSwapInt(this, UnsafeLock.stateOff, expected, state);
085    }
086
087    /**
088     * Performs CAS operation to replace the tail
089     *
090     * @param expected the expected tail
091     * @param tail     the new tail to set
092     */
093    private void updateTail(UnsafeLock.ThreadNode expected, UnsafeLock.ThreadNode tail) {
094        while (!UnsafeLock.UNSAFE.compareAndSwapObject(this, UnsafeLock.tailOff, expected, tail))
095            UnsafeLock.UNSAFE.putOrderedObject(this, UnsafeLock.tailOff, tail);
096    }
097
098    /**
099     * Linked-list node part of the waiting list of threads acquiring the lock
100     *
101     * @author AgentTroll
102     * @version 1.0
103     * @since 1.1
104     */
105    private class ThreadNode {
106        /** The previous node in the linked list */
107        @Getter private final UnsafeLock.ThreadNode previous;
108        /** The thread held by the node */
109        @Getter private final Thread                thread;
110
111        public ThreadNode(Thread thread) {
112            this.thread = thread;
113            this.previous = UnsafeLock.this.tail;
114        }
115    }
116}
