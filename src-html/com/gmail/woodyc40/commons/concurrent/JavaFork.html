017package com.gmail.woodyc40.commons.concurrent;
018
019import com.gmail.woodyc40.commons.Commons;
020import com.gmail.woodyc40.commons.io.*;
021import com.gmail.woodyc40.commons.misc.SerializableRunnable;
022import com.google.common.io.ByteStreams;
023import lombok.AccessLevel;
024import lombok.Getter;
025
026import javax.annotation.concurrent.NotThreadSafe;
027import java.io.File;
028import java.io.IOException;
029import java.lang.instrument.Instrumentation;
030import java.util.Enumeration;
031import java.util.jar.JarEntry;
032import java.util.jar.JarFile;
033
034/**
035 * Forks the JVM into a separate {@link java.lang.Process}, which runs the heartbeat for {@link
036 * com.gmail.woodyc40.commons.concurrent.ThreadPoolManager}
037 * <p>
038 * <p>NOT THREAD SAFE.</p>
039 *
040 * @author AgentTroll
041 * @version 1.0
042 * @since 1.0
043 */
044@NotThreadSafe
045public final class JavaFork {
046    /** The remote caller */
047    @Getter(AccessLevel.PACKAGE) private static Remotes         FROM_FORK;
048    /** The process started by the fork of the JVM */
049    private static                              Process         process;
050    /** The instrument */
051    @Getter private static                      Instrumentation instrument;
052    /** The remote receiver */
053    @Getter(AccessLevel.PACKAGE) private static Remotes         TO_FORK;
054
055    /** The security file */
056    private static File security;
057
058    private JavaFork() {} // Suppress instantiation
059
060    /**
061     * Beats at minecraft tick rate (20/sec). The while loop sleeps for 2 milliseconds and wakes to count the current
062     * time minus the starting time. If the elapsed time is larger than 50, or the time from the last beat exceeds 50,
063     * the ThreadPoolManager beats.
064     *
065     * @param args the command line arguments (none)
066     */
067    public static void main(String... args) {
068        long msec = 0L;
069        long current = System.currentTimeMillis();
070
071        while (true) {
072            long elapsed = System.currentTimeMillis() - current;
073            if (elapsed >= 50L || msec >= 50L) {
074                JavaFork.FROM_FORK.call(new SerializableRunnable<Void>() {
075                    private static final long serialVersionUID = 3355592265139036018L;
076
077                    @Override public Void run() {
078                        ThreadPoolManager.beat();
079                        return null;
080                    }
081                });
082                msec = 0L;
083            } else msec = elapsed;
084        }
085    }
086
087    /**
088     * Acquires the instance of {@link java.lang.instrument.Instrumentation}
089     *
090     * @param args       the arguments passed (none)
091     * @param instrument the instance of the instrument
092     */
093    public static void premain(String args, Instrumentation instrument) {
094        JavaFork.instrument = instrument;
095    }
096
097    /**
098     * Forks the JVM and starts a new task on this class
099     *
100     * @throws IOException when the process cannot be started (for some reason, possibly security restrictions)
101     */
102    public static void start() throws IOException {
103        JavaFork.genSecurityFiles();
104
105        JavaFork.process = new ProcessBuilder(Files.executable().getAbsolutePath(),
106                                              "-Djava.rmi.server.codebase=file:/" +
107                                              Files.jarFile().getAbsolutePath().replaceFirst("/", ""),
108                                              "-Djava.security.policy=" + JavaFork.security.getAbsolutePath(),
109                                              "-jar",
110                                              Files.jarFile().getName())
111                .directory(Files.pluginDir())
112                .inheritIO()
113                .start();
114
115        JavaFork.FROM_FORK = new Remotes("Fork2TPM");
116        JavaFork.TO_FORK = new Remotes(JavaFork.FROM_FORK);
117    }
118
119    /**
120     * Generate the security files in the parent folder
121     */
122    private static void genSecurityFiles() throws IOException {
123        File file = FileWrapper.handleFolder(Commons.getPlugin().getDataFolder());
124        final File src = new File(file, "src");
125        if (!src.exists()) {
126            src.mkdirs();
127            File jar = Files.jarFile();
128            JarFile jarFile = new JarFile(jar);
129            Enumeration<JarEntry> entries = jarFile.entries();
130            while (entries.hasMoreElements()) {
131                JarEntry entry = entries.nextElement();
132                if (entry.getName().endsWith(".java"))
133                    JavaFork.copy(jarFile, entry);
134            }
135        }
136
137        JavaFork.security = FileWrapper.handle(new File(file, "server.policy"));
138        final StringFileWriter writer = new StringFileWriter(JavaFork.security);
139
140        writer.writeLine("grant codeBase \"file:" + src.getAbsolutePath() + "\" {",
141                         "    permission java.security.AllPermission;",
142                         "};");
143
144        System.setProperty("java.security.policy", JavaFork.security.getAbsolutePath());
145        System.setProperty("java.rmi.server.codebase",
146                           "file:/" + src.getAbsolutePath().replaceFirst("/", ""));
147    }
148
149    /**
150     * Copy source file to data folder
151     *
152     * @param file  the originating jarfile the entry came from
153     * @param entry the entry to copy into the data folder
154     */
155    public static void copy(final JarFile file, final JarEntry entry) throws IOException {
156        final File newFile = new File(Commons.getPlugin().getDataFolder(), entry.getName());
157        com.google.common.io.Files.write(ByteStreams.toByteArray(file.getInputStream(entry)), newFile);
158    }
159
160    /**
161     * Stops the forked JVM upon exit of the main JVM (ie the CraftBukkit server)
162     */
163    public static void shutdown() {
164        if (JavaFork.process != null)
165            JavaFork.process.destroy();
166    }
167}
