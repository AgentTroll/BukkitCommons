017package com.gmail.woodyc40.commons.concurrent;
018
019import javax.annotation.concurrent.GuardedBy;
020import javax.annotation.concurrent.ThreadSafe;
021import java.util.Collection;
022import java.util.HashSet;
023import java.util.concurrent.*;
024
025/**
026 * Distributes a task amongst a list of {@link com.gmail.woodyc40.commons.concurrent.CountingExecutor}s, using a
027 * balancer and fallback ThreadPoolExecutors to assist with load estimates.
028 *
029 * @param <V> the return value of the Future representing the task submitted
030 * @author AgentTroll
031 * @version 1.0
032 * @since 1.0
033 */
034@ThreadSafe
035public class Distributor<V> implements Callable<Future<V>> {
036    /** The worker services */
037    @GuardedBy("services") private final Collection<CountingExecutor> services = new HashSet<>();
038    /** The task to submit */
039    private final              Callable<V>      task;
040    /** The fallback ExecutorService as the 2nd to last ditch executor for new tasks */
041    @GuardedBy("this") private CountingExecutor fallback;
042    /** The balancing ExecutorService used to handle the distribution */
043    @GuardedBy("this") private ExecutorService  balancer;
044
045    /**
046     * Builds a new distribution for the task provided
047     *
048     * @param task the task to distribute
049     */
050    public Distributor(Callable<V> task) {
051        this.task = task;
052    }
053
054    /**
055     * Sets the fallback executor to use if all workers are full
056     *
057     * @param fallback the fallback executor
058     * @return the instance the fallback was added to
059     */
060    public Distributor<V> setFallback(CountingExecutor fallback) {
061        synchronized (this) {
062            this.fallback = fallback;
063        }
064
065        return this;
066    }
067
068    /**
069     * Sets the balancing executor to submit the balancing task to
070     *
071     * @param balancer the balancing executor
072     * @return the instance the balancer was added to
073     */
074    public Distributor<V> setBalancer(ExecutorService balancer) {
075        synchronized (this) {
076            this.balancer = balancer;
077        }
078
079        return this;
080    }
081
082    /**
083     * Adds an executor to the consumer executor list
084     *
085     * @param service the worker executor
086     * @return the instance the worker was added to
087     */
088    public Distributor<V> addExecutor(CountingExecutor service) {
089        synchronized (this.services) {
090            this.services.add(service);
091        }
092
093        return this;
094    }
095
096    /**
097     * Selects the worker to use <p>The first step is to find the worker with the lowest work load</p>
098     * <p>
099     * <p>The second step checks the work load of the lowest worker, if it is within 5% of 100, the fallback is
100     * selected</p>
101     * <p>
102     * <p>However, if the fallback happens to be at 100% workload, then the task will be moved over to the balancer</p>
103     */
104    @Override public Future<V> call() throws Exception {
105        double usage = 100.0;
106        CountingExecutor executor = null;
107        synchronized (this.services) {
108            for (CountingExecutor service : this.services) {
109                if (service.percentUsed() < usage) {
110                    usage = service.percentUsed();
111                    executor = service;
112                }
113            }
114        }
115
116        synchronized (this) {
117            if (usage <= 100.0 && usage >= 95.0 && this.fallback.percentUsed() < 100.0) {
118                return this.fallback.submit(this.task);
119            } else if (this.fallback.percentUsed() == 100.0)
120                return this.balancer.submit(this.task);
121        }
122
123        return executor != null ? executor.submit(this.task) : null;
124    }
125}
