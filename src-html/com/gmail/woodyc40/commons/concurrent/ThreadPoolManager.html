017package com.gmail.woodyc40.commons.concurrent;
018
019import lombok.AccessLevel;
020import lombok.Getter;
021
022import javax.annotation.concurrent.ThreadSafe;
023import java.util.concurrent.*;
024import java.util.concurrent.atomic.AtomicInteger;
025
026/**
027 * Management of thread pooling and providing balancing support between the 2 load fixed length executors
028 *
029 * @author AgentTroll
030 * @version 1.0
031 * @since 1.0
032 */
033@ThreadSafe
034public final class ThreadPoolManager {
035    /** The beat count. Resets when the worker threads are recalculated */
036    private static final AtomicInteger beats = new AtomicInteger(0);
037
038    /** The first consumer executor */
039    private static final CountingExecutor LOAD  =
040            CountingExecutor.newCountingExecutor();
041    /** The second consumer executor */
042    private static final CountingExecutor LOAD0 =
043            CountingExecutor.newCountingExecutor();
044
045    /** Balancing executor */
046    private static final ExecutorService  BALANCER = Executors.newCachedThreadPool();
047    /** The fallback (last ditch) consumer executor */
048    private static final CountingExecutor FALLBACK = CountingExecutor.newCountingExecutor();
049
050    /** The remote caller */
051    @Getter(AccessLevel.PACKAGE) private static final Remotes TO_FORK   = JavaFork.getTO_FORK();
052    /** The remote receiver */
053    @Getter(AccessLevel.PACKAGE) private static final Remotes FROM_FORK = JavaFork.getFROM_FORK();
054
055    private ThreadPoolManager() {}
056
057    /**
058     * Checks the beat and recalculates on the 1000th beat from {@link com.gmail.woodyc40.commons.concurrent.JavaFork}
059     */
060    static void beat() {
061        if (ThreadPoolManager.beats.get() == 1000) {
062            ThreadPoolManager.LOAD.recalc();
063            ThreadPoolManager.LOAD0.recalc();
064            ThreadPoolManager.beats.set(0);
065        }
066
067        System.out.println("BEAT");
068    }
069
070    /**
071     * Submits a task for balancing on one of the thread pools
072     * <p>
073     * <p>The task MUST be designed asynchronously - there are no guarantees for thread safety, so synchronize,
074     * synchronize, synchronize!</p>
075     *
076     * @param task the task to execute concurrently
077     * @param <V>  the return type of the task
078     */
079    public static <V> void submit(Callable<V> task) {
080        Distributor<V> dist = new Distributor<>(task)
081                .addExecutor(ThreadPoolManager.LOAD)
082                .addExecutor(ThreadPoolManager.LOAD0)
083                .setBalancer(ThreadPoolManager.BALANCER)
084                .setFallback(ThreadPoolManager.FALLBACK);
085
086        ThreadPoolManager.BALANCER.submit(dist);
087    }
088
089    /**
090     * Ends the thread manager
091     */
092    public static void shutdown() {
093        ThreadPoolManager.LOAD.shutdownNow();
094        ThreadPoolManager.LOAD0.shutdownNow();
095        ThreadPoolManager.BALANCER.shutdownNow();
096        ThreadPoolManager.FALLBACK.shutdownNow();
097    }
098}
