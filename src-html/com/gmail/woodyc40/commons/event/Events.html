017package com.gmail.woodyc40.commons.event;
018
019import com.gmail.woodyc40.commons.Commons;
020
021import javax.annotation.concurrent.GuardedBy;
022import javax.annotation.concurrent.ThreadSafe;
023import java.util.*;
024import java.util.concurrent.LinkedBlockingDeque;
025
026/**
027 * The events utility for calling and registering their respective events through an asynchronous environment
028 *
029 * @author AgentTroll
030 * @version 2.0
031 * @since 1.0
032 */
033@ThreadSafe
034public final class Events {
035    /** The registered listeners */
036    @GuardedBy("Events.class") private static final Map<EventHandler, EventType> registered = new HashMap<>();
037    /** The event thread */
038    private static final                            Events.EventThread           THREAD     = new Events.EventThread();
039
040    static {
041        Events.THREAD.start();
042    }
043
044    private Events() {} // Suppress instantiation
045
046    /**
047     * Asynchronously register a handler
048     *
049     * @param handler the listener to register
050     * @throws NoEventAnnotationException when the handler is not annotated with a Handler annotation
051     */
052    public static void register(EventHandler handler) throws NoEventAnnotationException {
053        if (handler.getClass().getAnnotation(Handler.class) == null)
054            throw new NoEventAnnotationException(handler.getClass());
055
056        synchronized (Events.class) {
057            Events.registered.put(handler, handler.getClass().getAnnotation(Handler.class).value());
058        }
059    }
060
061    /**
062     * Asynchronously call the event, firing all of the listeners
063     *
064     * @param event the event to call
065     */
066    public static void call(CustomEvent event) {
067        Events.THREAD.call(event);
068    }
069
070    /**
071     * Stops the events from being registered and stops the event loop <p> <p>You should never ever call this.
072     * Nevertheless, this cannot be called by you anyways.</p>
073     */
074    public static void shutdown() throws SecurityException {
075        Package settings = Commons.getCaller(false);
076        if (settings == null)
077            throw new SecurityException("ACCESS DENIED: Not called from main class");
078        if (!settings.equals(Commons.class.getPackage()))
079            throw new SecurityException("ACCESS DENIED: You are not BukkitCommons");
080        Events.THREAD.interrupt();
081    }
082
083    /**
084     * The handler thread wrapper that calls and registers events in the event loop
085     *
086     * @author AgentTroll
087     * @version 1.0
088     * @since 1.1
089     */
090    private static class EventThread extends Thread {
091        /** The execution tasks */
092        private final Queue<Runnable> executor = new LinkedBlockingDeque<>();
093        /** If the thread is running, or naw */
094        private volatile boolean running;
095
096        /**
097         * Does the handling for the event call
098         *
099         * @param event the event to call
100         */
101        private static void handleEvent(CustomEvent event) {
102            synchronized (Events.class) {
103                for (Map.Entry<EventHandler, EventType> entry : Events.registered.entrySet())
104                    if (entry.getValue() == event.getClass().getAnnotation(Handler.class).value())
105                        entry.getKey().handle(event);
106            }
107        }
108
109        @Override public void run() {
110            while (this.running) {
111                Runnable runnable = this.executor.poll();
112                if (runnable == null) continue;
113                runnable.run();
114            }
115        }
116
117        @Override public void interrupt() {
118            super.interrupt();
119            this.running = false;
120        }
121
122        @Override public synchronized void start() {
123            super.start();
124            this.running = true;
125        }
126
127        /**
128         * Signals the thread to handle the available event listeners
129         *
130         * @param event the event to call
131         */
132        public void call(final CustomEvent event) {
133            this.executor.add(new Runnable() {
134                @Override public void run() {
135                    Events.EventThread.handleEvent(event);
136                }
137            });
138        }
139    }
140}
