017package com.gmail.woodyc40.commons.instrument.refs;
018
019import com.gmail.woodyc40.commons.collect.HashStructSet;
020import com.gmail.woodyc40.commons.instrument.experimental.ConstantNotFoundException;
021import com.gmail.woodyc40.commons.providers.UnsafeProvider;
022import sun.reflect.ConstantPool;
023
024import java.lang.reflect.Field;
025import java.lang.reflect.Method;
026import java.util.Collection;
027
028/**
029 * The reference to the entire constant pool itself
030 *
031 * @author AgentTroll
032 * @version 1.0
033 * @since 1.0
034 */
035public class PoolRef {
036    /** The reference to the constant references */
037    private final Collection<ConstantRef> refs = new HashStructSet<>();
038    /** The reference to the actual Constant Pool */
039    private final ConstantPool constantPool;
040    /** The holder of the constant pool */
041    private final Class<?>     clazz;
042
043    /**
044     * Builds a new constant pool reference from the javassist constant pool
045     *
046     * @param pool the constant pool
047     */
048    public PoolRef(Class<?> clazz, ConstantPool pool) {
049        this.constantPool = pool;
050        this.clazz = clazz;
051        for (int i = 0; i < this.constantPool.getSize(); i++) this.refs.add(this.read(i));
052    }
053
054    /**
055     * Adds an entry to the constant pool
056     *
057     * @param ref the reference to the entry to add
058     */
059    public void addRef(ConstantRef ref) {
060        this.refs.add(ref);
061        // TODO how are we gonna add this?
062    }
063
064    /**
065     * Read the value at the given index
066     *
067     * @param i the index to read the constant pool value
068     * @return the reference to the constant given at the index
069     */
070    public ConstantRef read(int i) {
071        Class<?> cls = null;
072        Double d = null;
073        Field fld = null;
074        Float f = null;
075        Integer in = null;
076        Long l = null;
077        String[] memberRefInfo = null;
078        Method method = null;
079        String s = null;
080        String utf = null;
081
082        try {
083            cls = this.constantPool.getClassAt(i);
084            d = this.constantPool.getDoubleAt(i);
085            fld = this.constantPool.getFieldAt(i);
086            f = this.constantPool.getFloatAt(i);
087            in = this.constantPool.getIntAt(i);
088            l = this.constantPool.getLongAt(i);
089            memberRefInfo = this.constantPool.getMemberRefInfoAt(i);
090            method = (Method) this.constantPool.getMethodAt(i);
091            s = this.constantPool.getStringAt(i);
092            utf = this.constantPool.getUTF8At(i);
093        } catch (Exception ignored) {
094        }
095
096        if (cls != null) return new ConstantRef(ConstantRef.Type.CLASS, cls, i);
097        else if (d != null) return new ConstantRef(ConstantRef.Type.DOUBLE, d, i);
098        else if (fld != null) return new ConstantRef(ConstantRef.Type.FIELD, fld, i);
099        else if (f != null) return new ConstantRef(ConstantRef.Type.FLOAT, f, i);
100        else if (in != null) return new ConstantRef(ConstantRef.Type.INTEGER, in, i);
101        else if (l != null) return new ConstantRef(ConstantRef.Type.LONG, l, i);
102        else if (memberRefInfo != null) return new ConstantRef(ConstantRef.Type.MEMBER_REF, memberRefInfo, i);
103        else if (method != null) return new ConstantRef(ConstantRef.Type.METHOD, method, i);
104        else if (s != null) return new ConstantRef(ConstantRef.Type.STRING, s, i);
105        else if (utf != null) return new ConstantRef(ConstantRef.Type.UTF8, utf, i);
106        else {
107            UnsafeProvider.exception(new ConstantNotFoundException(this.clazz, i));
108            return null;
109        }
110    }
111
112    /**
113     * Gets the maximum index of the wrapped constant pool
114     *
115     * @return the last index of which a constant entry can be found at
116     */
117    public int getSize() {
118        return this.constantPool.getSize();
119    }
120
121    /**
122     * Complete adding the values to the constant pool and push changes to the javassist long vector
123     */
124    public void finish() {
125
126    }
127}
