017package com.gmail.woodyc40.commons.misc;
018
019import lombok.Getter;
020
021import java.io.PrintStream;
022import java.util.ArrayList;
023import java.util.Collection;
024
025/**
026 * ASCII table wrapper that can be formatted and printed out to the console
027 *
028 * @author AgentTroll
029 * @version 1.0
030 * @since 1.0
031 */
032public class Table {
033    /** Table data rows */
034    private final Collection<Row> rowEntries = new ArrayList<>();
035    /** Represents the maximum lines to append in ASCII */
036    int[] max = new int[0];
037    /** Columns of data to support */
038    @Getter int columns;
039    /** The titles of the table */
040    private String[] names = { };
041
042    /**
043     * Makes a new row and appends it to the row collection
044     *
045     * @return the new row created
046     */
047    public Row createRow() {
048        Row row = new RowEntry(this);
049        this.rowEntries.add(row);
050        return row;
051    }
052
053    /**
054     * Appends a line to the table
055     */
056    public void appendSeperator() {
057        this.rowEntries.add(new Line());
058    }
059
060    /**
061     * The titles of each column on the table
062     * <p>
063     * <p>Unlimited slots are available, in fact, the amount of columns are set by this method.</p>
064     *
065     * @param names the titles
066     */
067    public void setNames(String... names) {
068        this.names = names.clone();
069
070        this.max = new int[this.columns = names.length];
071    }
072
073    /**
074     * Prints out the formatted table to the stream
075     *
076     * @param stream the stream to print out to
077     */
078    public void print(PrintStream stream) {
079        StringBuilder format = new StringBuilder();
080        format.append('|');
081
082        StringBuilder dashes = new StringBuilder();
083
084        for (int i1 = 0; i1 < this.max.length; i1++) {
085            int maximum = this.max[i1] > this.names[i1].length() ? this.max[i1] : this.names[i1].length();
086            format.append("%-").append(maximum).append('s');
087            format.append('|');
088
089            if (i1 == 0) dashes.append('+');
090            for (int i = 0; i < maximum; i++) {
091                dashes.append('-');
092                if (i == maximum - 1) dashes.append('+');
093            }
094        }
095
096        stream.println(dashes);
097        stream.printf(format.toString(), this.names);
098        System.out.println();
099        stream.println(dashes);
100
101        for (Row rowEntry : this.rowEntries) {
102            if (rowEntry instanceof Line) {
103                stream.println(dashes);
104                continue;
105            }
106            stream.printf(format.toString(), rowEntry.getEntries());
107            stream.print("\n");
108        }
109        stream.println(dashes);
110    }
111}
