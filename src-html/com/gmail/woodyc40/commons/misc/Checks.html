017package com.gmail.woodyc40.commons.misc;
018
019/**
020 * Shortcut methods to throw exceptions if condition is not met
021 *
022 * @author AgentTroll
023 * @version 1.0
024 * @since 1.0
025 */
026public final class Checks {
027    private Checks() {}
028
029    /**
030     * Throws an {@link IllegalArgumentException} if the field provided is {@code null}.
031     *
032     * @param nullable the object to check not {@code null}
033     * @param message  the message to send for the cause of the exception
034     * @throws IllegalArgumentException when a nullable value is {@code null}
035     */
036    public static void notNull(Object nullable, String message) {
037        if (nullable == null)
038            throw new IllegalArgumentException(message);
039    }
040
041    /**
042     * Throws an {@link IllegalArgumentException} objects provided are not equal to each other.
043     *
044     * @param val0 the first object to check against val1
045     * @param val1 the second comparison value to check against
046     * @throws IllegalArgumentException when the two values do not match
047     */
048    public static void match(Object val0, Object val1) {
049        if (!val0.equals(val1))
050            throw new IllegalArgumentException("The first value does not match the second");
051    }
052}
