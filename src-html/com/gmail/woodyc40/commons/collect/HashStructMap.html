017package com.gmail.woodyc40.commons.collect;
018
019import lombok.Getter;
020
021import java.util.*;
022
023/**
024 * Implementation of {@link java.util.Map} using {@link com.gmail.woodyc40.commons.collect.AbstractHashStruct}
025 *
026 * @param <K> the type of key to use
027 * @param <V> the type of value to use
028 * @author AgentTroll
029 * @version 1.0
030 * @see java.util.Map
031 * @since 1.0
032 */
033public class HashStructMap<K, V> implements Map<K, V> {
034    /** The delegating hash structure to perform base ops on */
035    @Getter protected final AbstractHashStruct<K, V> delegate;
036
037    /**
038     * Create a new map based on {@link com.gmail.woodyc40.commons.collect.HashStructMap} with starting size of 16
039     */
040    public HashStructMap() {
041        this(16);
042    }
043
044    /**
045     * Create a map based on {@link com.gmail.woodyc40.commons.collect.HashStructMap} with specified size
046     *
047     * @param size initial map size to use
048     */
049    public HashStructMap(final int size) {
050        this.delegate = new AbstractHashStruct<K, V>() {
051            @Override protected AbstractHashStruct.Node[] buckets() {
052                return new AbstractHashStruct.Node[size];
053            }
054        };
055    }
056
057    /**
058     * Create a map based on {@link com.gmail.woodyc40.commons.collect.HashStructMap} with initial entries
059     *
060     * @param master the parent map to copy the entries over
061     */
062    public HashStructMap(Map<? extends K, ? extends V> master) {
063        this(master.size());
064        this.putAll(master);
065    }
066
067    /**
068     * Create a map from a base struct map
069     *
070     * @param hashStruct the struct to use as a delegate
071     */
072    public HashStructMap(AbstractHashStruct<K, V> hashStruct) {
073        this.delegate = hashStruct;
074    }
075
076    @Override public int size() {
077        return this.delegate.getSize();
078    }
079
080    @Override public boolean isEmpty() {
081        return this.size() == 0;
082    }
083
084    @Override public boolean containsKey(Object o) {
085        return this.delegate.containsKey((K) o); // Let it throw ClassCastException if it has to
086    }
087
088    @Override public boolean containsValue(Object o) {
089        return this.delegate.containsValue((V) o);
090    }
091
092    @Override public V get(Object o) {
093        return this.delegate.get((K) o);
094    }
095
096    @Override public V put(K k, V v) {
097        return this.delegate.put(k, v);
098    }
099
100    @Override public V remove(Object o) {
101        return this.delegate.remove((K) o);
102    }
103
104    @Override public void putAll(Map<? extends K, ? extends V> map) {
105        for (Map.Entry<? extends K, ? extends V> entry : map.entrySet())
106            this.delegate.put(entry.getKey(), entry.getValue());
107    }
108
109    @Override public void clear() {
110        this.delegate.clear();
111    }
112
113    @Override public Set<K> keySet() {
114        Set<K> keys = new HashStructSet<>();
115        for (AbstractHashStruct<K, V>.Node bucket : this.delegate.getBuckets()) {
116            for (AbstractHashStruct<K, V>.Node last = bucket; last != null; last = last.getNext())
117                keys.add(last.getKey());
118        }
119        return keys;
120    }
121
122    @Override public Collection<V> values() {
123        Collection<V> values = new ArrayList<>();
124        for (AbstractHashStruct<K, V>.Node bucket : this.delegate.getBuckets()) {
125            for (AbstractHashStruct<K, V>.Node last = bucket; last != null; last = last.getNext())
126                values.add(last.getValue());
127        }
128        return values;
129    }
130
131    @Override public Set<Map.Entry<K, V>> entrySet() {
132        Set<Map.Entry<K, V>> keys = new HashStructSet<>();
133        for (AbstractHashStruct<K, V>.Node bucket : this.delegate.getBuckets()) {
134            for (AbstractHashStruct<K, V>.Node last = bucket; last != null; last = last.getNext())
135                keys.add(new AbstractMap.SimpleEntry<>(last.getKey(), last.getValue()));
136        }
137        return keys;
138    }
139}
