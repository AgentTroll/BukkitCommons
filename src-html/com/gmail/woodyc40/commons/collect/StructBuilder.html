017package com.gmail.woodyc40.commons.collect;
018
019import java.util.Map;
020import java.util.Set;
021
022/**
023 * Used for building new {@link com.gmail.woodyc40.commons.collect.HashStructMap}s and {@link
024 * com.gmail.woodyc40.commons.collect.HashStructSet}s using properties specified
025 *
026 * @author AgentTroll
027 * @version 1.0
028 * @since 1.0
029 */
030public class StructBuilder {
031    /** The hash strategy */
032    private AbstractHashStruct.HashStrategy hash    = AbstractHashStruct.HashStrategy.A_TROLL;
033    /** The initial size */
034    private int                             size    = 16;
035    /** The resizing threshold */
036    private int                             resize  = 14;
037    /** Weak reference key collection configuration */
038    private boolean                         weakKey = false;
039    /** Weak reference value collection configuration */
040    private boolean                         weakVal = false;
041    /** Whether or not to enable concurrency */
042    private boolean concurrent; // TODO
043
044    /**
045     * Sets the hash strategy
046     *
047     * @param strategy the strategy to use
048     * @return the instance of the builder it was added to
049     */
050    public StructBuilder hash(AbstractHashStruct.HashStrategy strategy) {
051        this.hash = strategy;
052        return this;
053    }
054
055    /**
056     * Sets the initial size
057     *
058     * @param size the size to set to
059     * @return the instance of the builder it was added to
060     */
061    public StructBuilder size(int size) {
062        this.size = size;
063        return this;
064    }
065
066    /**
067     * Sets the resize threshold
068     *
069     * @param resize the size to resize the map at, doubled each time it is resized
070     * @return the instance of the builder it was added to
071     */
072    public StructBuilder resize(int resize) {
073        this.resize = resize;
074        return this;
075    }
076
077    public StructBuilder weakKeys() {
078        this.weakKey = true;
079        return this;
080    }
081
082    public StructBuilder weakValues() {
083        this.weakVal = true;
084        return this;
085    }
086
087    /**
088     * Whether or not to use concurrent collections
089     *
090     * @param concurrent {@code true} to enable concurrency support
091     * @return the instance of the builder it was added to
092     */
093    public StructBuilder concurrent(boolean concurrent) {
094        this.concurrent = concurrent;
095        return this;
096    }
097
098    /**
099     * Builds the resulting map
100     *
101     * @param <K> the key to use
102     * @param <V> the value to use
103     * @return the map built with the specified properties
104     */
105    public <K, V> Map<K, V> buildMap() {
106        AbstractHashStruct<K, V> hashStruct = new AbstractHashStruct() {
107            {
108                this.setResizeThresh(StructBuilder.this.resize);
109            }
110
111            @Override protected AbstractHashStruct.Node[] buckets() {
112                return new AbstractHashStruct.Node[StructBuilder.this.size];
113            }
114
115            @Override protected AbstractHashStruct.HashStrategy hashStrategy() {
116                return StructBuilder.this.hash;
117            }
118        };
119
120        if (this.weakKey && this.weakVal)
121            return new WeakStructMap<>(WeakStructMap.WeakConfiguration.BOTH);
122        if (this.weakKey)
123            return new WeakStructMap<>(WeakStructMap.WeakConfiguration.KEYS);
124        if (this.weakVal)
125            return new WeakStructMap<>(WeakStructMap.WeakConfiguration.VALUES);
126        return new HashStructMap<>(hashStruct);
127    }
128
129    /**
130     * Builds the resulting set
131     *
132     * @param <E> the element type
133     * @return the set built with the specified properties
134     */
135    public <E> Set<E> buildSet() {
136        AbstractHashStruct<E, Object> hashStruct = new AbstractHashStruct() {
137            {
138                this.setResizeThresh(StructBuilder.this.resize);
139            }
140
141            @Override protected AbstractHashStruct.Node[] buckets() {
142                return new AbstractHashStruct.Node[StructBuilder.this.size];
143            }
144
145            @Override protected AbstractHashStruct.HashStrategy hashStrategy() {
146                return StructBuilder.this.hash;
147            }
148        };
149
150        // TODO weak implementations
151        return new HashStructSet<>(hashStruct);
152    }
153}
