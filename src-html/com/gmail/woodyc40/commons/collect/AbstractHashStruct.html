043@Getter
044public abstract class AbstractHashStruct<K, V> {
045    /** The entry storage */
046    private AbstractHashStruct<K, V>.Node[] buckets = this.buckets();
047
048    /** Amount of entries inserted */
049    private int size;
050    /** The threshold until the map resizes */
051    @Setter private int resizeThresh = 14;
052
053    /** The strategy employed to hash the keys on insertion */
054    @Setter private AbstractHashStruct.HashStrategy strategy = this.hashStrategy();
055

059     * Provider of the bucket array of nodes

061     * @return the initial value of the node array, including size

063    protected abstract AbstractHashStruct.Node[] buckets();

065    protected AbstractHashStruct.HashStrategy hashStrategy() {

071    /**
072     * Insertion operation on the structure
073     *
074     * @param k the key
075     * @param v the value
076     * @return the old value, or null if there is none
077     */
078    public final V put(K k, V v) {
079        int hash = this.posOf(k);
080        AbstractHashStruct<K, V>.Node toSet = this.search(k, hash);
081        V old = toSet.getValue();
082
083        toSet.setBucket(hash);
084        toSet.setValue(v);
085        return old;
086    }
087
088    /**
089     * Retrieves the value associated with the key provided
090     *
091     * @param k the key to get the value from
092     * @return the value associated with the key upon insertion, null if the key is not found, or the key is associated
093     * with a null value
094     */
095    public final V get(K k) {
096        int hash = this.posOf(k);
097        AbstractHashStruct<K, V>.Node node = this.loop(k, this.buckets[hash]);
098        if (node == null) return null;
099        else return node.getValue();
100    }
101
102    /**
103     * Removes an entry from the bucket array
104     *
105     * @param k the key which a value is associated to that will be removed from the map
106     * @return the old value associated with the key removed, null if it is not actually removed, the key is not found,
107     * or the value associated with the key is null
108     */
109    public final V remove(K k) {
110        AbstractHashStruct<K, V>.Node head = this.buckets[this.posOf(k)];
111        if (head == null) return null;
112        V old = head.getValue();
113
114        this.remove(k, head);
115        return old;
116    }
117
118    /**
119     * Removes all entries from the bucket array, setting the size to zero, along with the resizing threshold to 14.
120     */
121    public final void clear() {
122        this.buckets = new AbstractHashStruct.Node[16];
123        this.size = 0;
124        this.resizeThresh = 14;
125    }
126
127    /**
128     * Asks the bucket array if one of the entries has a key provided
129     *
130     * @param ky the key to check for in the bucket array
131     * @return {@code false} if the key is not found in the buckets, {@code true} if found
132     */
133    public final boolean containsKey(K ky) {
134        K k;
135        return (k = this.loop(ky, this.buckets[this.posOf(ky)]).getKey()) != null && k.equals(ky);
136    }
137
138    /**
139     * Asks the bucket array if one of the entries holds the value provided <p> <p>Note that this is an O(n)
140     * operation</p>
141     *
142     * @param v the value to check for in the bucket array
143     * @return {@code false} if the value is not found in the buckets, {@code true} if found
144     */
145    public final boolean containsValue(V v) {
146        for (AbstractHashStruct<K, V>.Node node : this.buckets)
147            if (node.getValue().equals(v))
148                return true;
149        return false;
150    }

172    /**
173     * Returns the position of the key via hashing and finding the hashed bucket
174     *
175     * @param key the key to use in hashing
176     * @return the position in the bucket array
177     */
178    private int posOf(K key) {
179        return this.strategy.hash(key, this.buckets.length);
180    }
181
182    /**
183     * Resizes the bucket array if necessary
184     */
185    private void checkSize() {
186        if (this.size == this.resizeThresh) {
187            int newLen = this.resizeThresh *= 2;
188
189            AbstractHashStruct<K, V>.Node[] resize = new AbstractHashStruct.Node[newLen];
190            System.arraycopy(this.buckets, 0, resize, 0, this.size);
191            this.buckets = resize;
192        }
193    }
194
195    /**
196     * Iterates through the linked list bucket index for the node containing the key
197     *
198     * @param k    the key to get the node from
199     * @param head the first node in the bucket
200     * @return the node that holds the key, null if not found
201     */
202    private AbstractHashStruct<K, V>.Node loop(K k, AbstractHashStruct<K, V>.Node head) {
203        AbstractHashStruct<K, V>.Node tail = head;
204        while (tail != null) {
205            if (tail.getKey().equals(k))
206                return tail;
207            tail = tail.getNext();
208        }
209
210        return null;
211    }
212
213    /**
214     * Performs a {@link #loop(Object, com.gmail.woodyc40.commons.collect.AbstractHashStruct.Node)}, returning the node
215     * if non-{@code null}, or creating a key if {@code null}
216     *
217     * @param k    the key to get the node from
218     * @param hash the position of the key
219     * @return the node associated with the key in the bucket array
220     */
221    private AbstractHashStruct<K, V>.Node search(K k, int hash) {
222        // Implementation:
223        // 1. Find the bucket
224        // 2. Loop to the last element in the bucket
225        // 3. If the last element isn't null, return it
226        // 4. Else, create a new key and return that
227
228        AbstractHashStruct<K, V>.Node head = this.buckets[hash];
229        AbstractHashStruct<K, V>.Node tail = this.loop(k, head);
230        if (tail != null) return tail;
231        else return this.create(k, null, hash);
232    }
233
234    /**
235     * Creates a node and links it with the previous, or sets the bucket position to the node if the first in the
236     * particular bucket
237     *
238     * @param k        the key to create the node with
239     * @param previous the previous node, can be {@code null}
240     * @param hash     the position to place the node in the bucket array
241     * @return the node created
242     */
243    private AbstractHashStruct<K, V>.Node create(K k, AbstractHashStruct<K, V>.Node previous, int hash) {
244        AbstractHashStruct<K, V>.Node node = new AbstractHashStruct<K, V>.Node(k, null, 0, null);
245        if (previous == null) {
246            this.buckets[hash] = node;
247            this.size++;
248            this.checkSize();
249            return node;
250        }
251
252        previous.setNext(node);
253        this.size++;
254        this.checkSize();
255        return node;
256    }
257
258    /**
259     * Removes a key from a bucket
260     *
261     * @param k    the key in the entry which will be removed
262     * @param head the first node in the bucket of the to-be removed entry
263     */
264    private void remove(K k, AbstractHashStruct<K, V>.Node head) {
265        AbstractHashStruct<K, V>.Node leading = head;
266        AbstractHashStruct<K, V>.Node tail = leading == null ? null : leading.getNext();
267
268        while (tail != null) {
269            if (tail.getKey().equals(k)) {
270                leading.setNext(tail.getNext());
271                this.size--;
272                return;
273            }
274
275            tail = tail.getNext();
276            leading = leading.getNext();
277        }
278
279        // If it reaches here, it can mean that
280        // leading != null && trailing == null or
281        // leading == null && trailing == null
282
283        if (leading != null && leading.getKey().equals(k))
284            this.buckets[leading.getBucket()] = null;
285
286        this.size--;
287    }

291    /**
292     * An enumeration of strategies for finding the position of an object key on a hashtable
293     *
294     * @author AgentTroll
295     * @version 1.0
296     * @see com.google.common.hash.Murmur3_32HashFunction
297     */
298    public enum HashStrategy {
299        /**
300         * Uses {@link com.google.common.hash.Murmur3_32HashFunction} to hash the values, and mod the length of buckets
301         */
302        MURMUR {
303            @Override public int hash(Object o, int i) {
304                // Implementation:
305                // 3 step process -
306                // Step 1: Hash the current hashCode of the key
307                // Step 2: The value may be negative, convert to unsigned long
308                // Step 3: Find the remainder by dividing by the amount of buckets
309                // The value of Step 3 is returned as the bucket index
310
311                int hash = AbstractHashStruct.HashStrategy.hasher.hashInt(o.hashCode()).asInt();
312                long convert = UnsafeProvider.normalize(hash);
313
314                return (int) (convert % (long) i);
315            }
316        },
317
318        /**
319         * The default {@link java.util.HashMap} hashing strategy, from OpenJDK
320         */
321        JAVA {
322            @Override public int hash(Object o, int i) {
323                int h = o.hashCode();
324                h ^= h >>> 20 ^ h >>> 12;
325                int bit = h ^ h >>> 7 ^ h >>> 4;
326
327                return bit & i - 1;
328            }
329        },
330
331        /**
332         * Custom hashing function that is faster than MURMUR and better distribution than JAVA
333         */
334        A_TROLL {
335            @Override public int hash(Object o, int i) {
336                long h = UnsafeProvider.normalize(o.hashCode());
337                h = (h >> 16 ^ h) * 0x33L;
338                h = (h >> 16 ^ h) * 0x33L;
339                h = h >> 16 ^ h;
340
341                return (int) (h % (long) i);
342            }
343        };
344
345        /** The hasher to use in {@link com.gmail.woodyc40.commons.collect.AbstractHashStruct.HashStrategy#MURMUR} */
346        private static final HashFunction hasher = Hashing.murmur3_32();
347
348        /**
349         * Hash the key for the length of the bucket array
350         *
351         * @param key the key to find the position of
352         * @param len the number of bucket positions available
353         * @return the index to put the key
354         */
355        public abstract int hash(Object key, int len);
356    }

384    /**
385     * The iterator which goes over all the keys represented by this structure
386     *
387     * @author AgentTroll
388     * @version 1.0
389     * @see java.util.Iterator
390     * @see com.gmail.woodyc40.commons.collect.AbstractIterator
391     * @since 1.0
392     */
393    protected class KeyIterator extends AbstractIterator<K> {
394        @Override protected Object[] allVals() {
395            int index = 0;
396            Object[] vals = new Object[AbstractHashStruct.this.getSize()];
397            for (AbstractHashStruct<K, V>.Node bucket : AbstractHashStruct.this.getBuckets()) {
398                AbstractHashStruct<K, V>.Node tail = bucket;
399                while (tail != null) {
400                    vals[index] = tail.getKey();
401                    index++;
402                    tail = tail.getNext();
403                }
404            }
405            return vals;
406        }
407
408        @Override protected void removeValue(K val) {
409            AbstractHashStruct.this.remove(val);
410        }
411    }
